    let mockTransaction: Transaction;
    let baseContext: TransferContext;

        mockDb.sequelize.query.mockResolvedValue([null, 1]);

        // Default: cross-hire not completed
        mockHasCrossHireCompleted.mockReturnValue(false);

    describe('validate()', () => {
        it('should pass when all required fields present', () => {
            expect(() => step.validate(baseContext)).not.toThrow();

        it('should throw when old contract id is missing', () => {
            const invalidContext = ({
                ...baseContext,
                contracts: {

                        peoContract: baseContext.contracts.old.peoContract,
                    },

            expect(() => step.validate(invalidContext)).toThrow();

            const invalidContext = ({
                ...baseContext,

    describe('execute()', () => {
        describe('Happy Path', () => {
            it('should create contract successfully', async () => {
                const result = await step.execute(baseContext, mockTransaction);

                expect(mockCreateContract).toHaveBeenCalled();
                expect(result.context.contracts.new).toBeDefined();
                expect(result.context.contracts.new?.id).toBe(999);
                expect(result.context.contracts.new?.HrisProfileId).toBe(444);
                expect(result.result).toBe(StepExecutionResult.CONTINUE);
                expect(result.context.stepState.completedSteps).toContain('CreateContract');

            it('should copy all employee data from old contract', async () => {
                await step.execute(baseContext, mockTransaction);

                expect(mockCreateContract).toHaveBeenCalledWith(

            it('should copy scope of work when available', async () => {
                await step.execute(baseContext, mockTransaction);

                expect(mockGetPeoScopeOfWorkByPeoContract).toHaveBeenCalledWith(123);
                expect(mockCreateContract).toHaveBeenCalledWith(
                    expect.objectContaining({
                        contractDetails: expect.objectContaining({
                            holidays: 10,
                            timeOffType: 'PTO',
                            scopeId: 'SCOPE-1',
                            jobScope: 'Engineering',
                            seniorityId: 'SENIOR-1',

                    true
                );

                await step.execute(baseContext, mockTransaction);

                    expect.objectContaining({

                            peoStartDate: new Date('2025-02-01'),

            it('should handle addressLine1 when present', async () => {
                await step.execute(baseContext, mockTransaction);

                expect(mockCreateContract).toHaveBeenCalledWith(
                    expect.objectContaining({
                        contractDetails: expect.objectContaining({
                            employeeAddress: expect.objectContaining({
                                street: '123 Main St',
                            }),
                        }),
                    }),
                    true
                );
            });

            it('should fallback to address field when addressLine1 is empty', async () => {
                const contextWithAddress = ({
                    ...baseContext,
                    contracts: {
                        old: {

                expect(mockCreateContract).toHaveBeenCalledWith(
                    expect.objectContaining({
                        contractDetails: expect.objectContaining({
                            employeeAddress: expect.objectContaining({

                    true

                                geoCode: 'NYC-001',
                            }),
                        }),
                    }),
                    true

            });

            it('should map employeeTypeCode F to Full-time', async () => {

                expect(mockCreateContract).toHaveBeenCalledWith(
                    expect.objectContaining({

                            employmentType: 'Full-time',
                        }),
                    }),
                    true
                );
            });

            it('should map employeeTypeCode P to Part-time', async () => {
                const contextWithPartTime = ({
                    ...baseContext,
                    contracts: {
                        old: {
                            ...baseContext.contracts.old,

                                ...baseContext.contracts.old.peoContract,
                                employeeTypeCode: 'P',
                            },
                        },
                    },
                } as unknown) as TransferContext;

                await step.execute(contextWithPartTime, mockTransaction);

                        contractDetails: expect.objectContaining({
                            employmentType: 'Part-time',
                        }),
                    }),
                    true
                );
            });

            it('should map employeeTypeCode S to Seasonal', async () => {
                const contextWithSeasonal = ({
                    ...baseContext,
                    contracts: {
                        old: {
                            ...baseContext.contracts.old,
                            peoContract: {
                                ...baseContext.contracts.old.peoContract,
                                employeeTypeCode: 'S',
                            },
                        },
                    },
                } as unknown) as TransferContext;

                await step.execute(contextWithSeasonal, mockTransaction);

                expect(mockCreateContract).toHaveBeenCalledWith(
                    expect.objectContaining({
                        contractDetails: expect.objectContaining({
                            employmentType: 'Seasonal',
                        }),
                    }),
                    true
                );
            });

            it('should default unknown employeeTypeCode to Full-time', async () => {
                const contextWithUnknownType = ({
                    ...baseContext,
                    contracts: {
                        old: {
                            ...baseContext.contracts.old,

                        },

                await step.execute(contextWithUnknownType, mockTransaction);

                            employmentType: 'Full-time',
                        }),
                    }),
                    true
                );
            });

            it('should include optional fields when present', async () => {
                await step.execute(baseContext, mockTransaction);

                expect(mockCreateContract).toHaveBeenCalledWith(

                        contractDetails: expect.objectContaining({
                            workHoursPerWeek: 40,
                            variableCompensations: 'Bonus: $5000',

                            fedFileStatus: 'M1',

            it('should use EntityTransfer as origin', async () => {
                await step.execute(baseContext, mockTransaction);

                expect(mockCreateContract).toHaveBeenCalledWith(
                    expect.objectContaining({
                        contractDetails: expect.objectContaining({

                    }),
                    true

            });

            it('should fetch HrisProfileId if not returned by createContract', async () => {

                    transaction: mockTransaction,
                    useMaster: true,

                expect(result.context.contracts.new?.HrisProfileId).toBe(444);
            });

            it('should pass transaction to createContract', async () => {
                await step.execute(baseContext, mockTransaction);

                expect(mockCreateContract).toHaveBeenCalledWith(
                    expect.objectContaining({
                        transaction: mockTransaction,
                    }),
                    true
                );
            });

            it('should pass creator, organization, and teamId to createContract', async () => {
                await step.execute(baseContext, mockTransaction);

                expect(mockCreateContract).toHaveBeenCalledWith(
                    expect.objectContaining({
                        creator: baseContext.initiator.creator,
                        organization: baseContext.initiator.organization,
                        teamId: 777,
                    }),
                    true
                );
            });

            it('should log contract creation with duration', async () => {
                await step.execute(baseContext, mockTransaction);

                expect(mockLogger.info).toHaveBeenCalledWith(
                    expect.objectContaining({
                        message: '[CreateContractStep] Contract created successfully',
                        transferId: 'transfer-1',
                        itemId: 'item-1',
                        newContractId: 999,
                        hrisProfileId: 444,
                        duration: expect.any(Number),
                    })
                );
            });

            it('should resolve jobTitle from jobCode and pass to createContract', async () => {
                await step.execute(baseContext, mockTransaction);

                expect(mockGetPEOJobCodesByEntityId).toHaveBeenCalledWith(200);
                expect(mockCreateContract).toHaveBeenCalledWith(
                    expect.objectContaining({
                        peopleFirstPayload: expect.objectContaining({
                            validatedData: expect.objectContaining({
                                jobCode: 'JOB-001',
                                jobTitle: 'Software Engineer',
                            }),
                        }),
                    }),
                    true
                );
            });

            it('should log successful jobTitle resolution', async () => {
                await step.execute(baseContext, mockTransaction);

                expect(mockLogger.info).toHaveBeenCalledWith(
                    expect.objectContaining({
                        message: '[CreateContractStep] Resolved jobTitle from jobCode',
                        jobCode: 'JOB-001',
                        jobTitle: 'Software Engineer',
                    })
                );
            });

            it('should throw when jobTitle not found in job master', async () => {
                mockGetPEOJobCodesByEntityId.mockResolvedValue([]);

                const promise = step.execute(baseContext, mockTransaction);
                await expect(promise).rejects.toThrow(/Job code JOB-001 not found/);
            });

            it('should throw when job code lookup fails', async () => {
                mockGetPEOJobCodesByEntityId.mockRejectedValue(new Error('API error'));

                await expect(step.execute(baseContext, mockTransaction)).rejects.toThrow('API error');
            });

            it('should throw when matched jobTitle is empty string', async () => {
                mockGetPEOJobCodesByEntityId.mockResolvedValue([{id: 'JOB-001', jobTitle: ''}] as any);

                const promise = step.execute(baseContext, mockTransaction);
                await expect(promise).rejects.toThrow(/Job code JOB-001 not found/);
            });
        });

            it('should throw when HrisProfile not found', async () => {
                mockDb.HrisProfile.findByPk.mockResolvedValue(null);

                await expect(step.execute(baseContext, mockTransaction)).rejects.toThrow('HrisProfile not found for contract 123');
            });

            it('should log warning when scope of work fails', async () => {
                const testError = new Error('Scope fetch failed');
                mockGetPeoScopeOfWorkByPeoContract.mockRejectedValue(testError);

                await step.execute(baseContext, mockTransaction);

                expect(mockLogger.warn).toHaveBeenCalledWith(
                    expect.objectContaining({
                        message: '[CreateContractStep] Failed to load scope of work',
                        oldContractId: 123,
                        error: testError,
                    })
                );

                const callArgs = mockCreateContract.mock.calls[0][0];
                expect(callArgs.contractDetails).not.toHaveProperty('holidays');
                expect(callArgs.contractDetails).not.toHaveProperty('timeOffType');
            });

            it('should log warning when geoCode fetch fails', async () => {
                const testError = new Error('GeoCode fetch failed');
                mockDb.Address.findByPk.mockRejectedValue(testError);

                await step.execute(baseContext, mockTransaction);

                expect(mockLogger.warn).toHaveBeenCalledWith(
                    expect.objectContaining({
                        message: '[CreateContractStep] Failed to get geoCode from address',
                        deelAddressId: 789,
                        error: testError,
                    })
                );

                const callArgs = mockCreateContract.mock.calls[0][0];
                expect(callArgs.contractDetails.employeeAddress).not.toHaveProperty('geoCode');
            });

            it('should log error and re-throw when createContract fails', async () => {
                const testError = new Error('Contract creation failed');
                mockCreateContract.mockRejectedValue(testError);

                await expect(step.execute(baseContext, mockTransaction)).rejects.toThrow('Contract creation failed');

                expect(mockLogger.error).toHaveBeenCalledWith(
                    expect.objectContaining({
                        message: '[CreateContractStep] Contract creation failed',
                        err: testError,
                        transferId: 'transfer-1',
                        itemId: 'item-1',
                    })
                );
            });

            it('should throw when HrisProfileId not found after refresh', async () => {
                mockCreateContract.mockResolvedValue({
                    id: 999,
                    oid: 'new-contract-oid',
                    HrisProfileId: undefined,
                    // eslint-disable-next-line @typescript-eslint/no-explicit-any -- Mock return value for createContract
                } as any);

                mockDb.Contract.findByPk.mockResolvedValue({
                    id: 999,
                    HrisProfileId: null,
                });

                await expect(step.execute(baseContext, mockTransaction)).rejects.toThrow('Created contract 999 has no HrisProfileId');
            });
        });
    });

    describe('rollback()', () => {
        describe('Rollback Flow', () => {
            it('should delete contract and all related records in order', async () => {
                const contextWithNewContract = ({
                    ...baseContext,
                    contracts: {
                        ...baseContext.contracts,
                        new: {id: 999, oid: 'new-contract-oid'},
                    },
                } as unknown) as TransferContext;

                mockDb.WorkStatement.destroy.mockResolvedValue(2);
                mockDb.UserContract.destroy.mockResolvedValue(1);
                mockDb.EmployeeOnboardingStep.destroy.mockResolvedValue(3);
                mockDb.ContractActionEvent.destroy.mockResolvedValue(5);
                mockDb.Guarantee.destroy.mockResolvedValue(0);
                mockDb.Contract.destroy.mockResolvedValue(1);

                await step.rollback(contextWithNewContract, mockTransaction);

                expect(mockDb.WorkStatement.destroy).toHaveBeenCalledWith({
                    where: {ContractId: 999},
                    transaction: mockTransaction,
                });
                expect(mockDb.UserContract.destroy).toHaveBeenCalledWith({
                    where: {contractId: 999},
                    transaction: mockTransaction,
                });
                expect(mockDb.EmployeeOnboardingStep.destroy).toHaveBeenCalledWith({
                    where: {contractId: 999},
                    transaction: mockTransaction,
                });
                expect(mockDb.ContractActionEvent.destroy).toHaveBeenCalledWith({
                    where: {contractId: 999},
                    transaction: mockTransaction,
                });
                expect(mockDb.Guarantee.destroy).toHaveBeenCalledWith({
                    where: {ContractId: 999},
                    transaction: mockTransaction,
                });
                expect(mockDb.Contract.destroy).toHaveBeenCalledWith({
                    where: {id: 999},
                    transaction: mockTransaction,
                });
            });

            it('should block rollback when cross-hire completed', async () => {
                // Mock the function to return true for this test
                mockHasCrossHireCompleted.mockReturnValue(true);

                const contextWithCrossHire = ({
                    ...baseContext,
                    contracts: {
                        ...baseContext.contracts,
                        new: {id: 999, oid: 'new-contract-oid'},
                    },
                    stepState: {
                        completedSteps: ['CrossHire'], // Use CrossHireStep.stepName value
                    },
                } as unknown) as TransferContext;

                await step.rollback(contextWithCrossHire, mockTransaction);

                expect(mockHasCrossHireCompleted).toHaveBeenCalledWith(contextWithCrossHire, 'CrossHire');
                expect(mockDb.WorkStatement.destroy).not.toHaveBeenCalled();
                expect(mockDb.Contract.destroy).not.toHaveBeenCalled();
                expect(mockDb.sequelize.query).not.toHaveBeenCalled();
                expect(mockLogger.error).toHaveBeenCalledWith(
                    expect.objectContaining({
                        message: '[CreateContractStep] Rollback skipped - cross-hire already completed',
                        newContractId: 999,
                        warning: 'Contract has PrismHR employee associated - manual cleanup required',
                    })
                );
            });

            it('should skip rollback when no contract created', async () => {
                await step.rollback(baseContext, mockTransaction);

                expect(mockDb.WorkStatement.destroy).not.toHaveBeenCalled();
                expect(mockLogger.warn).toHaveBeenCalledWith(
                    expect.objectContaining({
                        message: '[CreateContractStep] Rollback skipped - no contract to delete',
                    })
                );
            });

            it('should log deletion counts for each table', async () => {
                const contextWithNewContract = ({
                    ...baseContext,
                    contracts: {
                        ...baseContext.contracts,
                        new: {id: 999, oid: 'new-contract-oid'},
                    },
                } as unknown) as TransferContext;

                mockDb.WorkStatement.destroy.mockResolvedValue(2);
                mockDb.UserContract.destroy.mockResolvedValue(1);
                mockDb.EmployeeOnboardingStep.destroy.mockResolvedValue(3);
                mockDb.ContractActionEvent.destroy.mockResolvedValue(5);
                mockDb.Guarantee.destroy.mockResolvedValue(0);
                mockDb.Contract.destroy.mockResolvedValue(1);

                await step.rollback(contextWithNewContract, mockTransaction);

                expect(mockLogger.info).toHaveBeenCalledWith(
                    expect.objectContaining({
                        message: '[CreateContractStep] Contract hard deleted successfully during rollback',
                        summary: {
                            peoContractRollbackEventPublished: true,
                            workStatements: 2,
                            userContracts: 1,
                            onboardingSteps: 3,
                            actionEvents: 5,
                            guarantees: 0,
                        },
                    })
                );
            });

            it('should clear context.contracts.new after successful deletion', async () => {
                const contextWithNewContract = ({
                    ...baseContext,
                    contracts: {
                        ...baseContext.contracts,
                        new: {id: 999, oid: 'new-contract-oid'},
                    },
                } as unknown) as TransferContext;

                mockDb.WorkStatement.destroy.mockResolvedValue(2);
                mockDb.UserContract.destroy.mockResolvedValue(1);
                mockDb.EmployeeOnboardingStep.destroy.mockResolvedValue(3);
                mockDb.ContractActionEvent.destroy.mockResolvedValue(5);
                mockDb.Guarantee.destroy.mockResolvedValue(0);
                mockDb.Contract.destroy.mockResolvedValue(1);

                await step.rollback(contextWithNewContract, mockTransaction);

                expect(contextWithNewContract.contracts.new).toBeUndefined();
            });

            it('should log error and re-throw when rollback fails', async () => {
                const contextWithNewContract = ({
                    ...baseContext,
                    contracts: {
                        ...baseContext.contracts,
                        new: {id: 999, oid: 'new-contract-oid'},
                    },
                } as unknown) as TransferContext;

                const testError = new Error('Deletion failed');
                mockDb.WorkStatement.destroy.mockRejectedValue(testError);

                await expect(step.rollback(contextWithNewContract, mockTransaction)).rejects.toThrow('Deletion failed');

                expect(mockLogger.error).toHaveBeenCalledWith(
                    expect.objectContaining({
                        message: '[CreateContractStep] Rollback failed',
                        err: testError,
                        newContractId: 999,
                    })
                );
            });
        });
    });
