    /**
     * Get payrollSettingsId by legalEntityId and ICPId
     * Matches the pattern used in afterCommit when creating PEO contracts
     */
    private async _getPayrollSettingsIdForLegalEntityAndIcp({
        legalEntityId,
        icpId,
    }: {
        legalEntityId: number;
        icpId: number;
    }): Promise<string | undefined> {
        const payrollSettings = await this.payrollSettingsService.getByLegalEntityId({
            legalEntityId,
        });

        // Find the active payroll settings for this ICP
        const matchingSettings = Array.isArray(payrollSettings)
            ? payrollSettings.find((ps: {ICPId?: number; isActive?: boolean}) => ps.ICPId === icpId && ps.isActive !== false)
            : payrollSettings;

        return matchingSettings?.id;
    }

    /**
     * Resolve jobTitle from jobCode, with fallback to jobCode if resolution fails
     */
    private async _resolveJobTitleFromJobCode({
        jobCode,
        legalEntityId,
    }: {
        jobCode: string;
        legalEntityId: number;
    }): Promise<string> {
        try {
            const resolved = await this._validateAndGetJobTitleCode(jobCode, legalEntityId);
            if (resolved.jobTitle) {
                return resolved.jobTitle;
            }
        } catch (error) {
            this.log.warn({
                message: '[_resolveJobTitleFromJobCode] Failed to resolve jobTitle from jobCode, using jobCode as fallback',
                jobCode,
                legalEntityId,
                error,
            });
        }
        // Fallback: use jobCode as jobTitle if resolution failed
        return jobCode;
    }

    async _validateAndGetJobTitleCode(jobTitle, entityId) {
        try {
            const clientCodes = await peoClientMasterService.getPEOJobCodesByEntityId(entityId);

            const matchJobCode = clientCodes.find((jc) => jc.id === jobTitle || jc.jobTitle === jobTitle);

            if (matchJobCode) {
                return {
                    jobCode: matchJobCode.id,
                    jobTitle: matchJobCode.jobTitle,
                };
            }
        } catch (error) {
            this.log.error({message: 'Failed getting the job codes', error});
        }

        return {
            jobCode: undefined,
            jobTitle,
        };
    }

    /**
     * Creates employment for a contract that is missing one.
     * Used by tech_ops endpoint for fixing entity transfer contracts.
     *
     * All data is fetched from the NEW contract and its peo_contracts record.
     * No need to reference the old contract - data was already copied during transfer.
     * startDate is read from peoContract.peoStartDate - no parameter needed.
     *
     * @param contractOid - The contract OID to create employment for
     * @returns Object with success status and employment ID
     */
    async createMissingEmploymentForContract({
        contractOid,
    }: {
        contractOid: string;
    }): Promise<{success: boolean; message: string; employmentId?: string; skipped?: boolean}> {
        // 1. Check if employment already exists (idempotent)
        const existingEmployments = await EMSEmploymentsService.searchEmployments({
            contractOids: [contractOid],
            includePayrollSettings: true,
        });

        if (existingEmployments && existingEmployments.length > 0) {
            return {
                success: true,
                skipped: true,
                message: `Employment already exists for contract ${contractOid}`,
                employmentId: existingEmployments[0].id,
            };
        }

        // 2. Fetch contract with PEO data (contractorId is the Profile.id)
        const contract = await this.getContractByOid({
            deelContractOid: contractOid,
            additionalAttributes: ['id', 'oid', 'country', 'teamId', 'invitedClientEmail', 'clientLegalEntityId', 'HrisProfileId', 'contractorId'],
            addPEOData: true,
        });

        if (!contract) {
            throw new HttpError.NotFound(`Contract not found: ${contractOid}`);
        }

        const peoContract = contract.peoContract;
        if (!peoContract) {
            throw new HttpError.NotFound(`PEO contract not found for: ${contractOid}`);
        }

        // 3. Validate peoStartDate exists
        if (!peoContract.peoStartDate) {
            throw new HttpError.BadRequest(`PEO contract has no peoStartDate: ${contractOid}`);
        }

        // Format startDate as YYYY-MM-DD
        const startDate = this._formatDate(peoContract.peoStartDate);

        // 4. Get profile and employee data
        // Contract.contractorId is the Profile.id
        if (!contract.contractorId) {
            throw new HttpError.NotFound(`Contract has no contractorId (Profile): ${contractOid}`);
        }

        // Get HrisProfile for email fallback
        const hrisProfile = contract.HrisProfileId
            ? await this.db.HrisProfile.findByPk(contract.HrisProfileId, {
                  attributes: ['id', 'email'],
                  useMaster: false,
              })
            : null;

        // Get Employee from Employment microservice using contractorId (Profile.id)
        const employee = await employmentService.getEmployeeByProfileId(contract.contractorId);

        if (!employee) {
            throw new HttpError.NotFound(`Employee not found for profile: ${contract.contractorId}`);
        }

        // 5. Get ICP (PrismHR)
        const prismIcp = await this.db.ICP.findFirst({
            attributes: ['id'],
            where: {name: 'PrismHR'},
            useMaster: false,
        });

        if (!prismIcp) {
            throw new HttpError.NotFound('PrismHR ICP not found');
        }

        // 5b. Get payrollSettingsId by legalEntityId and ICPId (same as afterCommit pattern)
        const payrollSettingsId = contract.clientLegalEntityId
            ? await this._getPayrollSettingsIdForLegalEntityAndIcp({
                  legalEntityId: contract.clientLegalEntityId,
                  icpId: prismIcp.id,
              })
            : undefined;

        this.log.info({
            message: '[createMissingEmploymentForContract] Resolved payrollSettingsId',
            contractOid,
            payrollSettingsId,
            clientLegalEntityId: contract.clientLegalEntityId,
            icpId: prismIcp.id,
        });

        // 6. Resolve jobTitle from jobCode
        if (!peoContract.jobCode) {
            throw new HttpError.BadRequest(`jobTitle could not be resolved - contract has no jobCode: ${contractOid}`);
        }

        const jobTitle = await this._resolveJobTitleFromJobCode({
            jobCode: peoContract.jobCode,
            legalEntityId: contract.clientLegalEntityId,
        });

        // 7. Create employment (matching afterCommit pattern with payrollSettingsId)
        this.log.info({
            message: '[createMissingEmploymentForContract] Creating missing employment for contract',
            contractOid,
            jobTitle,
            startDate,
            payrollLegalEntityId: contract.clientLegalEntityId,
            payrollSettingsId,
        });

        if (!contract.country) {
            throw new HttpError.BadRequest(`Contract missing country field: ${contractOid}`);
        }

        const employment = await employmentService.createPEOEmployment({
            contractOid,
            country: contract.country,
            state: peoContract.stateCode,
            clientTeamId: contract.teamId,
            // @ts-expect-error Incorrect types in createPEOEmployment
            rate: String(peoContract.payRate),
            scale: peoContract.payMethod === PEOPayMethodEnum.HOURLY ? 'HOUR' : 'YEAR',
            email: contract.invitedClientEmail || hrisProfile?.email,
            // @ts-expect-error Incorrect types in createPEOEmployment
            startDate,
            employeeKey: employee.id,
            employeeId: contract.contractorId,
            jobTitle,
            ...(contract.clientLegalEntityId ? {payrollLegalEntityId: contract.clientLegalEntityId} : {}),
            ICPId: prismIcp.id,
            gpContractId: contract.id,
            ...(payrollSettingsId ? {payrollSettingsId} : {}),
        });

        if (!employment) {
            throw new Error('Failed to create employment - service returned null');
        }

        this.log.info({
            message: '[createMissingEmploymentForContract] Employment created successfully',
            contractOid,
            employmentId: employment.id,
        });

        return {
            success: true,
            message: `Employment created successfully for contract ${contractOid}`,
            employmentId: employment.id,
        };
    }

    async updateWorkLocationOnEffectiveDate({contractOid, payload, profile}) {
        const {
            workLocationUpdateEffectiveDate,
            workingLocationId,
            reason: updateReason,
            additionalDetail: reasonAdditionalInfo,
            hasDifferentWorkLocationAndAddress,
        } = payload;

        const workLocation = await peoWorkLocationService.getWorkLocation(workingLocationId);
        const workLocationCode = workLocation.locationId;

        const contract = await this.getContractByOid({
            deelContractOid: contractOid,
            addPEOData: true,
            additionalAttributes: ['teamId', 'timezone', 'oid', 'HrisProfileId', 'effectivePlainDate', 'clientLegalEntityId'],
            transaction: null,
        });

        const plainWorkLocationEffectiveDate = workLocationUpdateEffectiveDate || moment().tz(contract.timezone).format('YYYY-MM-DD');
        const workLocationEffectiveDateIsToday = plainWorkLocationEffectiveDate === moment().tz(contract.timezone).format('YYYY-MM-DD');

        const legalEntity = await this.db.LegalEntity.findFirst({
            attributes: ['publicId'],
            where: {
                id: contract.clientLegalEntityId,
            },
            useMaster: false,
        });

        await peoWorkLocationService.validateWorkLocation(workLocation, contract.clientLegalEntityId);

        await this._createUserEventForPrismRequests(contract.clientLegalEntityId, workLocationCode, null, contract.id, profile.id);

        const workLocationIsInUnderReview =
            (
                await peoPrismResourceRequestService.getPrismResourceRequestByLegalEntityId({
                    legalEntityId: legalEntity.publicId,
                    prismRequestStatus: PrismResourceRequestStatus.UNDER_REVIEW,
                    resource: PrismResourceRequestResource.WORK_LOCATION,
                    description: workLocationCode,
                    useMaster: true, //temp solution
                })
            )?.length > 0;

        if (workLocationEffectiveDateIsToday && workLocationIsInUnderReview) {
            throw new HttpError(400, 'Not able to update contract: The Work Location is in Under Review please choose other Effective Date');
        }

        if (contract.peoContract.workLocationCode !== workLocationCode) {
            await peoContractMissingUpdateService.createOrUpdatePeoContractMissingUpdate({
                deelContractId: contract.id,
                type: PeoContractMissingUpdateType.WORK_LOCATION,
                updateRequestedAt: plainWorkLocationEffectiveDate,
                updatedValue: workLocationCode,
                oldValue: contract.peoContract.workLocationCode,
                deelUpdaterProfileId: profile.id,
                updateReason,
                additionalInfo: reasonAdditionalInfo,
            });

            await this.updatePEOContract({
                deelContractId: contract.id,
                workLocationCodeEffectiveDate: plainWorkLocationEffectiveDate,
                workLocationCode,
                hasDifferentWorkLocationAndAddress,
            });

            // Sync updated work location to HRIS labels
            try {
                await peoWorkLocationService.syncPeoWorkLocationToHrisLabels({deelContractOid: contractOid});
            } catch (error) {
                this.log.error({message: 'Failed to sync work location to HRIS labels during work location update', error, contractOid});
            }
        }
    }

    async _createUserEventForPrismRequests(legalEntityId, workLocationCode, jobCode, contractId, profileId) {
        let contractHasUnderReviewRequest = false;
        let underReviewWorkLocation = null;
        let underReviewPosition = null;

        const legalEntity = await this.db.LegalEntity.findFirst({
            attributes: ['publicId'],
            where: {
                id: legalEntityId,
            },
            useMaster: false,
        });

        const underReviewRequest = await peoPrismResourceRequestService.getPrismResourceRequestByLegalEntityId({
            legalEntityId: legalEntity.publicId,
            prismRequestStatus: PrismResourceRequestStatus.UNDER_REVIEW,
            useMaster: true,
        });

        if (workLocationCode) {
            underReviewWorkLocation = underReviewRequest.find(
                (request) => request.resource === PrismResourceRequestResource.WORK_LOCATION && request.description === workLocationCode
            );
        }

        if (jobCode) {
            underReviewPosition = underReviewRequest.find(
                (request) => request.resource === PrismResourceRequestResource.POSITION && request.description === jobCode
            );
        }

        if (underReviewWorkLocation) {
            await peoUserEventService.createPeoUserEvent({
                profileId,
                eventType: PeoUserEventTypes.WORK_LOCATION_REQUESTED,
                eventDate: moment().toDate(),
                description: underReviewWorkLocation.description,
                resourceId: underReviewWorkLocation.publicId,
                deelContractId: contractId,
            });

            contractHasUnderReviewRequest = true;
        }

        if (underReviewPosition) {
            await peoUserEventService.createPeoUserEvent({
                profileId,
                eventType: PeoUserEventTypes.POSITION_REQUESTED,
                eventDate: moment().toDate(),
                description: underReviewPosition.description,
                resourceId: underReviewPosition.publicId,
                deelContractId: contractId,
            });

            contractHasUnderReviewRequest = true;
        }

        return Boolean(contractHasUnderReviewRequest);
    }

    async _checkIfContractHasPrismResourceRequest(legalEntityId, workLocationCode, jobCode) {
        const legalEntity = await this.db.LegalEntity.findFirst({
            attributes: ['publicId'],
            where: {
                id: legalEntityId,
            },
            useMaster: false,
        });

        const underReviewRequest = await peoPrismResourceRequestService.getPrismResourceRequestByLegalEntityId({
            legalEntityId: legalEntity.publicId,
            prismRequestStatus: PrismResourceRequestStatus.UNDER_REVIEW,
            useMaster: true, //temp solution
        });

        const underReviewWorkLocation = underReviewRequest.some(
            (request) => request.resource === PrismResourceRequestResource.WORK_LOCATION && request.description === workLocationCode
        );

        const underReviewPosition = underReviewRequest.some(
            (request) => request.resource === PrismResourceRequestResource.POSITION && request.description === jobCode
        );

        const contractHasUnderReviewRequest = underReviewWorkLocation || underReviewPosition;

        return Boolean(contractHasUnderReviewRequest);
    }

    async #getCurrentGPSalaryInfo(contractId, date = moment().format('YYYY-MM-DD')) {
        // https://wiki.deel.network/i/12427#h-1-through-the-backend-gpcontractdatachanges-service
        // eslint-disable-next-line @typescript-eslint/no-var-requires -- avoid circular dependency
        const {gpContractDataChangesService} = require('../globalPayroll/contract_data_changes/gp_contract_data_changes_service');

        const [activeSalary] = await gpContractDataChangesService.Salary.getActiveDataForContract({contractId, atDate: date});
        if (!activeSalary || !activeSalary.data) {
            throw new Error(`No active salary data found for contract ${contractId} at date ${date}`);
        }
        const {rate, scale, payCurrency} = activeSalary.data;
        const {effectiveDate} = activeSalary;

        return {
            salary: rate,
            scale,
            effectiveDate,
            payCurrency,
        };
    }

    mapPayScaleForGPDataChanges(payMethod) {
        if (['HOUR', 'YEAR'].includes(payMethod)) {
            return payMethod;
        }
        // default to 'YEAR' if payMethod is not provided
        return payMethod === PEOPayMethodEnum.HOURLY ? 'HOUR' : 'YEAR';
    }

    payRateEffectiveDate(peoStartDate, effectiveDate) {
        if (!peoStartDate && !effectiveDate) return moment().format('YYYY-MM-DD');
        if (effectiveDate && peoStartDate) {
            return moment.max(moment(peoStartDate), moment(effectiveDate)).format('YYYY-MM-DD');
        }
        return effectiveDate || peoStartDate;
    }

    /**
     * Fetches and validates an existing HrisProfile for entity transfer scenarios.
     * Ensures the profile exists and belongs to the correct organization.
     *
     * @param hrisProfileId - The ID of the HrisProfile to fetch
     * @param organizationId - The organization ID to validate against
     * @param transaction - Optional database transaction
     * @returns The HrisProfile if found and valid, null if hrisProfileId is not provided
     * @throws Error if profile not found or belongs to different organization
     */
    async #fetchAndValidateHrisProfileForEntityTransfer(hrisProfileId: number | undefined, organizationId: number, transaction?: Transaction) {
        if (!hrisProfileId) {
            return null;
        }

        const existingHrisProfile = await this.db.HrisProfile.findByPk(hrisProfileId, {
            attributes: ['id', 'workerId', 'externalId', 'oid', 'userId', 'email', 'organizationId'],
            transaction,
            useMaster: Boolean(transaction),
        });

        if (!existingHrisProfile) {
            throw new Error(`HrisProfile not found with id: ${hrisProfileId}`);
        }

        if (existingHrisProfile.organizationId !== organizationId) {
            throw new Error(
                `HrisProfile ${hrisProfileId} belongs to organization ${existingHrisProfile.organizationId}, but contract is being created for organization ${organizationId}`
            );
        }

        this.log.info({
            message: 'Reusing existing HrisProfile for entity transfer',
            hrisProfileId: existingHrisProfile.id,
            organizationId: existingHrisProfile.organizationId,
        });

        return existingHrisProfile;
    }

    async postSalaryChangeToGPDataChanges({
        profileId,
        contractId,
        contractOid,
        legalEntityId,
        salary,
        payMethod,
        peoStartDate,
        effectiveDate,
        transaction,
        changelogContext = {},
        asyncContext = {},
    }: Record<string, unknown> & {
        profileId: number;
        contractId: number;
        contractOid: string;
        legalEntityId: number;
        salary?: number;
        payMethod?: string;
        peoStartDate?: string;
        effectiveDate?: string;
        transaction?: Transaction;
        changelogContext?: Record<string, string | number | undefined | unknown>;
        asyncContext?: Record<string, unknown>;
    }) {
        // eslint-disable-next-line @typescript-eslint/no-var-requires -- avoid circular dependency
        const {gpContractDataChangesService} = require('../globalPayroll/contract_data_changes/gp_contract_data_changes_service');
        try {
            effectiveDate = this.payRateEffectiveDate(peoStartDate, effectiveDate);

            if (!contractId) throw new Error('Missing contractId');
            if (!effectiveDate) throw new Error('Missing effectiveDate');

            if (!salary || !payMethod) {
                try {
                    const {salary: currentGPSalary, scale: currentGPSalaryScale} = await this.#getCurrentGPSalaryInfo(contractId, effectiveDate);
                    if (!salary) {
                        salary = currentGPSalary;
                    }
                    if (!payMethod) {
                        payMethod = currentGPSalaryScale;
                    }
                } catch (err) {
                    this.log.error(
                        {
                            err,
                            contractId,
                            effectiveDate,
                            salary,
                            payMethod,
                            peoStartDate,
                        },
                        "Active salary or scale for the contract doesn't exist. Either its a new contract or info doesn't exist"
                    );
                }
            }

            if (!payMethod) throw new Error('Unable to determine payMethod for salary update');

            const change = {
                effectiveDate,
                changeData: {
                    rate: String(salary),
                    scale: this.mapPayScaleForGPDataChanges(payMethod),
                },
                createdBy: profileId,
            };
            const enableModificationReasonCall = featureFlagsProvider.getFlags()?.peo?.enableModificationReasonCall;

            if (enableModificationReasonCall && Object.keys(changelogContext).length > 0 && changelogContext.updateReason) {
                const {actorEmail, organizationId, hrisProfileId, updateReason, contractOid} = changelogContext;
                const changelogContextService = new ChangelogContextService({db: this.db, logger: this.log});
                const fields: FieldChangeIdentifier[] = [
                    {
                        objectId: `${contractOid}_MAIN`,
                        idempotencyKey: 'COMPENSATION_RATE',
                        afterValue: String(salary),
                    },
                ];
                await changelogContextService.setContext({
                    actorEmail: String(actorEmail),
                    organizationId: organizationId as number,
                    domain: ChangelogContextDomain.PEO,
                    url: asyncContext?.url as string,
                    subject: {
                        hrisProfileId: hrisProfileId as number,
                    },
                    fields,
                    effectiveDate,
                    editReasonOid: updateReason as string,
                    transaction,
                });
            }
            await gpContractDataChangesService.Salary.addChanges(contractId, [change], transaction ? {transaction} : {});

            if (isPEOProratedSalaryEnabled(legalEntityId)) {
                await peoBuildPayrollReportUponContractChangePublisher.publishEvent({contractOid, effectiveDate, legalEntityId});
            }

            this.log.info({message: 'Successfully synced salary change', contractId, effectiveDate, salary, payMethod, peoStartDate});
        } catch (error) {
            this.log.error({message: 'Failed to sync salary change', error, contractId, effectiveDate, salary, payMethod, peoStartDate});
            throw new Error(`Error syncing salary change: ${error.message}`);
        }
    }

    async createContract(
        {contractDetails, creator, teamId, contractEventLogger, organization, ip, transitionId, draftQuoteId, transaction, peopleFirstPayload},
        bypassExistentEmailValidation = false
    ) {
        // 1. Validate PEO application (skip if peopleFirstPayload exists - already validated upstream)
        if (!peopleFirstPayload) {
            const peoApplication = await this.db.Organization.getPeoApplication(organization.id);
            if (!peoApplication) {
                throw new Error('The organization has no PEO application');
            }
        }

        const {
            employeeFirstName: employeeFirstAndMiddleName,
            employeeLastName,
            employeeEmail,
            employeeWorkEmail,
            employmentState,
            employmentType,
            employeeNationality,
            jobTitle: jobTitleInput,
            scope,
            currency,
            cycle = 'calendar-month',
            customFields,
            employeeAddress: addressData,
            clientLegalEntityId,
            benefitsGroup,
            salary,
            variableCompensations,
            workingLocationId,
            holidays,
            timeOffType,
            departmentId,
            orgStructures,
            hrisJobPositionId,
            payMethod,
            workHoursPerWeek,
            workerId,
            externalId,
            origin,
            scopeId,
            jobScope,
            seniorityId,
            county,
            payrollSettingsId,
            hrisIntegrationProviderId,
            seniorityDate,
            baseContractOid,
            gender,
            ssn,
            birthDate,
            maritalStatus,
            citizenStatus,
            payGroup,
            fedFileStatus,
            hrisProfileId,
        } = contractDetails;

        // 2. Check for duplicate contracts (skip if peopleFirstPayload exists - already validated upstream)
        if (!peopleFirstPayload) {
            const existPEOContractActiveForEmployee = await this._existsPEOContractActiveForEmployee({employeeEmail}, {transaction});
            if (existPEOContractActiveForEmployee && !bypassExistentEmailValidation) {
                throw new Error(`The user (${employeeEmail}) already has a PEO contract active`);
            }
        }

        const isRegularCreation = !origin || origin === PeoContractOrigin.Regular;
        const isMassOnboardingCreation = origin && origin === PeoContractOrigin.MassOnboarding;
        const isHrisIntegrationCreation = !isRegularCreation && !isMassOnboardingCreation;

        // 3. Use validated jobTitle/jobCode if provided, otherwise validate now
        let jobTitle = peopleFirstPayload?.validatedData?.jobTitle || null;
        let jobCode = peopleFirstPayload?.validatedData?.jobCode || null;

        if (!jobTitle && !jobCode && jobTitleInput) {
            ({jobTitle, jobCode} = await this._validateAndGetJobTitleCode(jobTitleInput, clientLegalEntityId));
        }

        // Included this logic from HRIS Integration, only positions created on Prism should be considered in the creation flow
        if (isHrisIntegrationCreation && !jobCode) {
            jobTitle = null;
        }

        if (creator) {
            creator.timezone = getValidUSTimezone(creator.timezone);
        }

        let workLocationCode = null;
        if (isRegularCreation || workingLocationId) {
            const workLocation = await peoWorkLocationService.getPEOWorkLocation(workingLocationId);
            workLocationCode = workLocation.locationId;
            await peoWorkLocationService.validateWorkLocation(workLocation, clientLegalEntityId);
        }

        if (!workLocationCode) {
            //this code doesn't return obsolete work location => we don't need to validate it
            workLocationCode = await peoWorkLocationService.getWorkLocationCodeByDeelLegalEntityAndState(clientLegalEntityId, addressData.state);
        }

        const {peoStartDate: peoStartDateRaw, startDate: contractStartDateRaw, endDate: contractEndDate} = contractDetails;

        const peoStartDate = this._formatDate(peoStartDateRaw, creator?.timezone),
            contractStartDate = this._formatDate(contractStartDateRaw, creator?.timezone),
            peoEndDate = this._formatDate(contractEndDate, creator?.timezone);

        const [employeeFirstName, employeeMiddleName] = employeeFirstAndMiddleName.split(' ');
        const employeeLegalName = `${employeeFirstAndMiddleName} ${employeeLastName}`;

        const workStatement = {
            cycleScale: cycle,
            scale: 'custom',
            firstCycleDay: 1,
            effectiveDate: peoStartDate,
            effectivePlainDate: peoStartDate,
            completionDate: peoEndDate,
            currency,
            ...(scope ? {scope} : {}),
        };

        const hrisProfileData = {
            firstName: employeeFirstName?.slice?.(0, 100),
            lastName: employeeLastName?.slice?.(0, 100),
            email: employeeEmail,
            ...(employeeNationality ? {employeeNationality} : {}),
        };

        // eslint-disable-next-line @typescript-eslint/no-var-requires -- avoid circular dependency
        const {createContract} = require('../contracts/create');

        const existedUserByEmail = await this.db.User.findFirst({
            attributes: ['id'],
            where: {
                normalizedEmail: normalizeEmail(employeeEmail),
            },
            useMaster: Boolean(transaction),
            raw: true,
            transaction,
        });

        const existingHrisProfile =
            peopleFirstPayload?.hrisProfile || (await this.#fetchAndValidateHrisProfileForEntityTransfer(hrisProfileId, organization.id, transaction));

        const contract = await createContract({
            creator,
            ...(teamId ? {teamId} : {}),
            clientLegalEntityId,
            contractData: {
                name: jobTitle ? `${employeeLegalName} - ${jobTitle}` : employeeLegalName,
                firstSign: 'client',
                country: 'US',
                state: employmentState,
                type: CONTRACT_TYPES.PEO,
                ...(customFields ? {customFields} : {}),
                currency: 'USD',
                jobTitleName: jobTitle ?? undefined,
                data: {additionalInfo: {payMethod, ...(gender ? {gender: mapPEOGenderToHrisGender(gender)} : {})}},
                transitionId,
                draftQuoteId,
            },
            workStatement,
            hrisProfileData: {
                firstName: employeeFirstName?.slice?.(0, 100),
                lastName: employeeLastName?.slice?.(0, 100),
                email: employeeEmail,
                ...(employeeWorkEmail ? {workEmail: employeeWorkEmail} : {}),
                ...(workerId ? {workerId} : {}),
                ...(externalId ? {externalId} : {}),
            },
            ip,
            contractEventLogger,
            organization,
            userId: existedUserByEmail?.id,
            transaction,
            ...(existingHrisProfile ? {peopleFirstPayload: {hrisProfile: existingHrisProfile}} : {}),
        });

        // For entity transfer: if we passed an existing profile but it wasn't used, set it explicitly
        if (existingHrisProfile && !contract.HrisProfileId) {
            this.log.info({
                message: 'Explicitly setting HrisProfileId for entity transfer',
                contractId: contract.id,
                hrisProfileId: existingHrisProfile.id,
            });
            await this.db.Contract.update({HrisProfileId: existingHrisProfile.id}, {where: {id: contract.id}, transaction});
            contract.HrisProfileId = existingHrisProfile.id;
        }

        const hrisProfile = await this.db.HrisProfile.findFirst({
            where: {id: contract.HrisProfileId},
            attributes: ['workerId', 'externalId', 'oid', 'userId'],
            transaction,
        });
        const existedUser = existedUserByEmail || (hrisProfile && hrisProfile.userId ? {id: hrisProfile.userId} : null);

        // Sidecar
        const payload = {
            contractId: contract?.oid,
            jobTitle,
            jobScope: scope,
            workStartDate: peoStartDate,
            effectiveDate: peoStartDate,
            workCompletionDate: peoEndDate,
        };
        await this.saveWorkRecordsSidecar({
            contractId: contract?.id,
            payload,
            transaction,
            employmentType,
            workHoursPerWeek,
            seniorityId,
            withContractUpdates: true,
        });

        // 5. Validate and enrich address with geoCode (if not already present)
        if ((!addressData.geoCode || addressData.geoCode === '') && addressData.zip) {
            const matchedAddress = await peoLocationService.validateAndGetEntityGeoCode({
                city: addressData.city,
                geoCode: addressData.geoCode,
                zipCode: addressData.zip,
            });

            addressData.geoCode = matchedAddress.geoCode;
        }

        const profileAddress = await this.db.Address.create(addressData, {transaction});

        const existedProfilePeo = existedUser
            ? await this._getProfilePEOByUser(
                  {
                      userId: existedUser.id,
                  },
                  {transaction}
              )
            : null;

        const profile =
            existedProfilePeo ||
            (await this.db.Profile.create(
                {
                    profileType: this.db.Profile.PROFILE_TYPES_ENUM.peo_employee,
                    username: employeeEmail.split('@')[0],
                    firstName: hrisProfileData.firstName,
                    ...(employeeMiddleName && {middleName: employeeMiddleName}),
                    lastName: hrisProfileData.lastName,
                    lastOrganizationId: organization.id,
                    lastTeam: teamId,
                    timezone: creator.timezone,
                    ...(jobTitle ? {jobTitle} : {}),
                    AddressId: profileAddress.id,
                    ...(existedUser ? {userId: existedUser.id} : {}),
                },
                {transaction}
            ));

        if (existedUser?.id) {
            await this.db.Profile.update(
                {
                    profileType: this.db.Profile.PROFILE_TYPES_ENUM.peo_employee,
                    username: employeeEmail.split('@')[0],
                    firstName: hrisProfileData.firstName,
                    ...(employeeMiddleName && {middleName: employeeMiddleName}),
                    lastName: hrisProfileData.lastName,
                    lastOrganizationId: organization.id,
                    ...(teamId ? {lastTeam: teamId} : {}),
                    ...(jobTitle ? {jobTitle} : {}),
                    AddressId: profileAddress.id,
                    timezone: creator.timezone,
                    email: normalizeEmail(employeeEmail),
                    // @ts-expect-error Values exist on the model Profile but not able to detect
                    kyc: this.db.Profile.KYC_STATUS_ENUM.REJECTED,
                    currentOnboardingStep: this.db.Profile.ONBOARDING_STEPS_ENUM.USER_REGISTERED,
                    currentOnboardingStepDate: moment().toDate(),
                },
                {
                    where: {
                        id: profile.id,
                    },
                    transaction,
                }
            );
        }

        const employeeAddress = await this.db.Address.create(addressData, {transaction});

        const employeeData = {
            profileId: profile.id,
            firstName: hrisProfileData.firstName,
            ...(employeeMiddleName && {middleName: employeeMiddleName}),
            lastName: hrisProfileData.lastName,
            ...(employeeNationality ? {nationality: employeeNationality} : {}),
            addressId: employeeAddress.id,
        };

        const employee = await this.upsertEmployeeDataForPayroll(employeeData);

        if (!employee) {
            throw new Error('Failed to create employee');
        }

        // 6. Setup HrisProfile (skip if peopleFirstPayload exists - already done in WorkersService.createPerson())
        // WorkersService.createPerson() already: creates address, sets startDate, updates nationalities, links address
        let hrisProfileAddress = peopleFirstPayload?.hrisProfileAddress || null;
        if (!hrisProfileAddress) {
            hrisProfileAddress = await this.db.Address.create(addressData, {transaction});

            await this.db.HrisProfile.update(
                {
                    startDate: contractStartDate,
                    ...(existedUser ? {userId: existedUser.id} : {}),
                },
                {
                    where: {id: contract.HrisProfileId},
                    transaction,
                }
            );

            if (employeeNationality) {
                await this.db.HrisProfileNationalities.updateNationalities(contract.HrisProfileId, [employeeNationality], transaction);
            }

            await this.db.HrisProfileAddresses.createOrUpdateHomeAddress(contract.HrisProfileId, hrisProfileAddress.id, transaction);
        }

        const finalContract = await this._setupFinalContract({contract, creator, employeeEmail, clientLegalEntityId, teamId, profile, transaction});
        contract.clientLegalEntityId = clientLegalEntityId;

        contract.hris = {workerId: hrisProfile.workerId, externalId: hrisProfile.externalId, oid: hrisProfile.oid};

        // Only update HRIS profile fields if NOT coming from people-first flow
        if (!peopleFirstPayload && (departmentId || orgStructures || hrisJobPositionId)) {
            // eslint-disable-next-line @typescript-eslint/no-var-requires -- avoid circular dependency
            const {updateHrisProfileFields} = require('../hris/profiles');

            const hrisProfile = await this.db.HrisProfile.findFirst({
                where: {
                    id: contract.HrisProfileId,
                },
                transaction,
            });

            await updateHrisProfileFields({
                updaterProfile: creator,
                profileId: creator.id,
                organizationId: organization.id,
                hrisJobPositionId,
                hrisProfile,
                departmentId,
                orgStructures,
                transaction,
            });
        }

        await transaction.afterCommit(async () => {
            const peoContract = await this.createPEOContract({
                firstName: hrisProfileData.firstName,
                lastName: hrisProfileData.lastName,
                peoStartDate,
                gender,
                origHireDate: contractStartDate,
                lastHireDate: contractStartDate,
                deelEntityId: String(clientLegalEntityId),
                deelContractId: contract.id,
                deelContractOid: contract.oid,
                prismEmployeeId: String(contract.id),

                deelAddressId: hrisProfileAddress.id,
                ...(hrisProfileAddress.zip ? {zipCode: hrisProfileAddress.zip} : {}),
                city: hrisProfileAddress.city,
                stateCode: hrisProfileAddress.state,
                addressLine1: hrisProfileAddress.street,

                employeeStatusCode: this.mapEmployeeStatus('Active'),
                ...(employmentType ? {employeeTypeCode: this.mapEmployeeTypeCode(employmentType)} : {}),

                payRate: String(salary),
                payPeriod: payMethod === PEOPayMethodEnum.HOURLY ? 'H' : 'Y',
                ...(payMethod ? {payMethod} : {payMethod: PEOPayMethodEnum.SALARY}),
                ...(workLocationCode ? {workLocationCode} : {}),
                ...(benefitsGroup ? {benefitsGroup} : {}),

                ...(variableCompensations ? {variableCompensations} : {}),

                ...(jobCode || jobTitle ? {jobCode: jobCode || jobTitle} : {}),
                ...(workHoursPerWeek ? {workHoursPerWeek} : {}),

                ...(origin ? {origin} : {origin: PeoContractOrigin.Regular}),
                ...(county ? {county} : {}),
                ...(hrisIntegrationProviderId ? {hrisIntegrationProviderId} : {}),
                ...(seniorityDate ? {seniorityDate, isRehire: true} : {}),
                ...(baseContractOid ? {baseContractOid} : {}),

                // Personal information required for cross-hire
                ...(ssn ? {ssn} : {}),
                ...(birthDate ? {birthDate} : {}),
                ...(maritalStatus ? {maritalStatus} : {}),
                ...(citizenStatus ? {citizenStatus} : {}),
                ...(payGroup ? {payGroup} : {}),
                ...(fedFileStatus ? {fedFileStatus} : {}),
            });

            await this._createUserEventForPrismRequests(clientLegalEntityId, workLocationCode, jobCode || jobTitle, contract.id, creator.id);

            await OnboardingPublishersHooks(this.db.Sequelize, this.db.sequelize).onStepEvaluationTriggered({
                organizationId: organization.id,
                legalEntityId: clientLegalEntityId,
            });

            await this.db.EmployeeOnboardingStep.initPEOOnboarding({
                profile,
                contract,
            });

            const isNewState = await this.isContractInNewState(hrisProfileAddress.state, contract.oid, clientLegalEntityId, transaction);
            const hasFirstPayrollProcessed = await this.hasFirstPayrollProcessedForEntity(clientLegalEntityId, transaction);

            if (isNewState && hasFirstPayrollProcessed) {
                await peoOpsWorkbenchService.createTaskForContractInNewState(
                    hrisProfileAddress.country,
                    hrisProfileAddress.state,
                    contract.oid,
                    clientLegalEntityId
                );
            }

            if (holidays || timeOffType || scopeId || seniorityId) {
                await peoScopeOfWorkService.upsertPeoScopeOfWork({
                    peoContractId: peoContract.id,
                    ...(holidays ? {holidays} : {}),
                    ...(timeOffType ? {timeOffType} : {}),
                    ...(scopeId ? {scopeId} : {}),
                    ...(!scopeId && jobScope ? {jobScope} : {}),
                    ...(seniorityId ? {seniorityId} : {}),
                });
            }
            const prismIcp = await this.db.ICP.findFirst({
                attributes: ['id'],
                where: {
                    name: 'PrismHR',
                },
                useMaster: false,
            });

            // Add this condition to be able to create employment for the Hris integration contract, since jobTitle is required, when the user finishes the Hris integration contract creation, the jobTitle is updated.
            if (!jobTitle && isHrisIntegrationCreation) {
                jobTitle = jobTitleInput;
            }

            const employment = await employmentService.createPEOEmployment({
                contractOid: contract.oid,
                country: contract.country,
                state: employmentState,
                clientTeamId: teamId,
                // @ts-expect-error Incorrect types in createPEOEmployment
                rate: String(salary),
                scale: payMethod === PEOPayMethodEnum.HOURLY ? 'HOUR' : 'YEAR',
                email: employeeEmail.toLowerCase(),
                // @ts-expect-error Incorrect types in createPEOEmployment
                startDate: peoStartDate,
                employeeKey: employee.id,
                employeeId: profile.id,
                ...(jobTitle ? {jobTitle} : {}),

                payrollLegalEntityId: clientLegalEntityId,
                ICPId: prismIcp.id,
                // Will allow us to reuse most of the GP flows for payroll
                gpContractId: contract.id,
                payrollSettingsId,
            });

            if (isPEOEmploymentEventEnabled()) {
                await peoEmploymentEventService.publishEmploymentCreated({
                    id: contract.oid,
                    startDate: peoStartDate,
                    employeeName: employeeLegalName,
                    ...(peoEndDate ? {endDate: peoEndDate} : {}),
                    status: PEOEmploymentStatus.ONBOARDING,
                    endsWithOffCycle: false,
                });
            }

            if (!employment) {
                throw new Error('Failed to create employment');
            }

            if (isRegularCreation) {
                this.log.info({message: 'Updating salary information in GP contract data changes as it is a regular contract flow'});
                await this.postSalaryChangeToGPDataChanges({
                    profileId: creator.id,
                    contractId: contract.id,
                    contractOid: contract.oid,
                    legalEntityId: clientLegalEntityId,
                    salary,
                    payMethod: payMethod ?? PEOPayMethodEnum.SALARY,
                    peoStartDate,
                });
            }

            const timeTrackingContractPayload = {
                contractType: CONTRACT_TYPES.PEO,
                isHourly: payMethod === PEOPayMethodEnum.HOURLY,
                isEmbeddedPayroll: false,
                hourlyRate: salary,
                legalEntityId: clientLegalEntityId,
                organizationId: organization.id,
                country: contract.country,
                ICPId: prismIcp.id,
                employmentType,
                scheduleId: contractDetails.timeTracking?.scheduleId,
            };

            const shouldCreateTimeTrackingContract = await timeTrackingService.isAvailableForTimeTracking({
                contractType: timeTrackingContractPayload.contractType,
                isHourly: timeTrackingContractPayload.isHourly,
                organizationId: timeTrackingContractPayload.organizationId,
                countryCode: timeTrackingContractPayload.country,
                // @ts-expect-error TODO: fix this
                flsaOvertimeStatus: timeTrackingContractPayload.flsaOvertimeStatus,
                scheduleId: timeTrackingContractPayload.scheduleId,
            });

            if (shouldCreateTimeTrackingContract) {
                await timeTrackingService.createTimeTrackingContract({
                    oid: contract.oid,
                    ...timeTrackingContractPayload,
                });
            }

            try {
                await this.evaluatePeoVipTagAssignment({
                    contract,
                    peoContract,
                });
            } catch (error) {
                // do not throw error here to avoid breaking the contract creation flow
                this.log.error({message: 'Failed to evaluate PEO VIP tag assignment', error, contractId: contract.id});
            }

            if (isRegularCreation) {
                await this.sendPEOContractInvitation(
                    {...hrisProfileData, profileId: profile.id, userId: existedUser?.id},
                    organization,
                    finalContract,
                    hrisProfileAddress
                );
            } else {
                await this.db.Contract.update(
                    {
                        // We use this status to define that the contract is waiting for prism resource request
                        status: CONTRACT_STATUSES.UNDER_REVIEW,
                    },
                    {
                        where: {
                            id: contract.id,
                        },
                    }
                );
            }

            // Sync PEO work location to HRIS labels for SSoT compatibility
            try {
                await peoWorkLocationService.syncPeoWorkLocationToHrisLabels({deelContractOid: contract.oid, useMaster: true});
            } catch (err) {
                this.log.error({message: 'Failed to sync work location to HRIS labels during contract creation', err, contractOid: contract.oid});
                // Don't throw error to avoid breaking the contract creation flow
            }
        });

        return {
            ...contract,
            ...finalContract.dataToUpdateFinalContract,
            Contractor: {
                ...profile,
            },
        };
    }

    async hasFirstPayrollProcessedForEntity(clientLegalEntityId: number, transaction: Transaction) {
        const usePMS = PmsEmsSelectionService.isPMSEnabled({legalEntityId: clientLegalEntityId, models: ['Groups', 'Cycles']});

        try {
            if (usePMS) {
                const [result] = await this.db.sequelize.query<{hasCompletedPayrollEvent: boolean}>(
                    sql`
                    SELECT EXISTS (
                        SELECT 1
                        FROM payroll_management.cycles cy
                        INNER JOIN payroll_management."groups" gr ON cy.group_id = gr.id
                        WHERE gr.legal_entity_id = :legalEntityId
                            AND cy.state = 'COMPLETED'
                    ) AS "hasCompletedPayrollEvent";`,
                    {
                        type: this.db.Sequelize.QueryTypes.SELECT,
                        replacements: {legalEntityId: clientLegalEntityId},
                        raw: true,
                        useMaster: Boolean(transaction),
                    }
                );
                return Boolean(result?.hasCompletedPayrollEvent);
            }

            const [result] = await this.db.sequelize.query<{hasCompletedPayrollEvent: boolean}>(
                sql`
                SELECT EXISTS (
                SELECT 1
                FROM employment.payroll_events pe
                INNER JOIN employment.payroll_settings ps ON pe.payroll_settings_id = ps.id
                INNER JOIN public."LegalEntities" le ON ps.payroll_legal_entity_id = le.id
                WHERE le.id = :legalEntityId
                    AND pe.action_state = 'COMPLETE'
                ) AS "hasCompletedPayrollEvent";`,
                {
                    type: this.db.Sequelize.QueryTypes.SELECT,
                    replacements: {legalEntityId: clientLegalEntityId},
                    raw: true,
                    useMaster: Boolean(transaction),
                }
            );
            return Boolean(result?.hasCompletedPayrollEvent);
        } catch (err) {
            this.log.error({message: 'Failed to check if first payroll is processed for legal entity', err, legalEntityId: clientLegalEntityId, usePMS});
            // Don't throw error here to avoid breaking the contract creation flow
            return true;
        }
    }

    async updateAgreement({contractOid, payload, profile}) {
        const {
            teamId,
            jobTitle: jobTitleInput,
            salary,
            peoStartDate: peoStartDateRaw,
            startDate: contractStartDateRaw,
            effectiveDate,
            updateReason,
            reasonAdditionalInfo,
            scopeId,
            jobScope,
            seniorityId,
            employmentType,
            payMethod,
            workHoursPerWeek,
            timeTracking,
        } = payload;

        const contract = await this.getContractByOid({
            deelContractOid: contractOid,
            addPEOData: true,
            additionalAttributes: [
                'teamId',
                'timezone',
                'oid',
                'HrisProfileId',
                'effectiveDate',
                'effectivePlainDate',
                'initialEffectiveDate',
                'clientLegalEntityId',
                'data',
                'createdAt',
                'contractorId',
                'organizationId',
            ],
        });

        const plainEffectiveDate = effectiveDate || moment().tz(contract.timezone).format('YYYY-MM-DD');
        const effectiveDateIsToday = effectiveDate === moment().tz(contract.timezone).format('YYYY-MM-DD');

        const {jobTitle, jobCode} = await this._validateAndGetJobTitleCode(jobTitleInput, contract.clientLegalEntityId);

        const newJob = jobTitleInput || jobTitle || jobCode;
        if (newJob) {
            await this._createUserEventForPrismRequests(contract.clientLegalEntityId, null, newJob, contract.id, profile.id);
        }

        const peoStartDate = this._formatDate(peoStartDateRaw, contract.timezone);
        const contractStartDate = this._formatDate(contractStartDateRaw, contract.timezone);
        const oldPeoStartDate = this._formatDate(contract.effectiveDate);
        const oldContractStartDate = this._formatDate(contract.initialEffectiveDate);
        this._validateDatesUpdate(peoStartDate, oldPeoStartDate, contractStartDate, oldContractStartDate);

        await this.db.sequelize.transaction(async (transaction) => {
            const legalEntity = await this.db.LegalEntity.findFirst({
                attributes: ['publicId'],
                where: {
                    id: contract.clientLegalEntityId,
                },
                useMaster: false,
            });

            const newPosition = jobCode || jobTitle;

            const positionInUnderReview =
                (
                    await peoPrismResourceRequestService.getPrismResourceRequestByLegalEntityId({
                        legalEntityId: legalEntity.publicId,
                        prismRequestStatus: PrismResourceRequestStatus.UNDER_REVIEW,
                        resource: PrismResourceRequestResource.POSITION,
                        description: newPosition,
                    })
                )?.length > 0;

            if (effectiveDateIsToday && positionInUnderReview) {
                throw new HttpError(400, 'Not be able to update contract the Position is in Under review please choose other Effectivedate');
            }

            if (teamId !== contract.teamId) {
                await peoContractMissingUpdateService.createOrUpdatePeoContractMissingUpdate({
                    deelContractId: contract.id,
                    type: PeoContractMissingUpdateType.GROUP,
                    updateRequestedAt: effectiveDate,
                    updatedValue: String(teamId),
                    oldValue: String(contract.teamId),
                    deelUpdaterProfileId: profile.id,
                    updateReason,
                    additionalInfo: reasonAdditionalInfo,
                });
            }

            if (String(salary) !== String(contract.peoContract.payRate)) {
                await peoContractMissingUpdateService.createOrUpdatePeoContractMissingUpdate({
                    deelContractId: contract.id,
                    type: PeoContractMissingUpdateType.SALARY,
                    updateRequestedAt: effectiveDate,
                    updatedValue: String(salary),
                    oldValue: String(contract.peoContract.payRate),
                    deelUpdaterProfileId: profile.id,
                });

                const slackMessagePayload = {
                    employeeName: `${contract.peoContract.firstName} ${contract.peoContract.lastName}`,
                    oid: contract.oid,
                    retroactive: moment(effectiveDate) < moment(),
                    oldSalary: String(contract.peoContract.payRate),
                    newSalary: String(salary),
                    effectiveDate: moment(effectiveDate).format('LL'),
                };

                await slackDoAction({type: 'PEO_SALARY_EDIT_REQUEST', payload: slackMessagePayload});
            }

            if (newPosition !== contract.peoContract.jobCode) {
                await peoContractMissingUpdateService.createOrUpdatePeoContractMissingUpdate({
                    deelContractId: contract.id,
                    type: PeoContractMissingUpdateType.JOB_TITLE,
                    updateRequestedAt: effectiveDate,
                    updatedValue: newPosition,
                    oldValue: contract.peoContract.jobCode,
                    deelUpdaterProfileId: profile.id,
                    updateReason,
                    additionalInfo: reasonAdditionalInfo,
                });
            }

            const employeeTypeCode = this.mapEmployeeTypeCode(employmentType);

            if (employeeTypeCode !== contract.peoContract.employeeTypeCode) {
                await peoContractMissingUpdateService.createOrUpdatePeoContractMissingUpdate({
                    deelContractId: contract.id,
                    type: PeoContractMissingUpdateType.EMPLOYMENT_TYPE,
                    updateRequestedAt: effectiveDate,
                    updatedValue: String(employeeTypeCode),
                    oldValue: contract.peoContract.employeeTypeCode,
                    deelUpdaterProfileId: profile.id,
                    updateReason,
                    additionalInfo: reasonAdditionalInfo,
                });
            }

            if (workHoursPerWeek !== contract.peoContract.workHoursPerWeek) {
                await peoContractMissingUpdateService.createOrUpdatePeoContractMissingUpdate({
                    deelContractId: contract.id,
                    type: PeoContractMissingUpdateType.WORK_HOURS,
                    updateRequestedAt: effectiveDate,
                    updatedValue: String(workHoursPerWeek),
                    oldValue: String(contract.peoContract.workHoursPerWeek ?? 0),
                    deelUpdaterProfileId: profile.id,
                    updateReason,
                    additionalInfo: reasonAdditionalInfo,
                });
            }

            if (payMethod !== contract.peoContract.payMethod) {
                await peoContractMissingUpdateService.createOrUpdatePeoContractMissingUpdate({
                    deelContractId: contract.id,
                    type: PeoContractMissingUpdateType.PAY_METHOD,
                    updateRequestedAt: effectiveDate,
                    updatedValue: payMethod,
                    oldValue: contract.peoContract.payMethod,
                    deelUpdaterProfileId: profile.id,
                    updateReason,
                    additionalInfo: reasonAdditionalInfo,
                });
            }

            await this.db.Contract.update(
                {
                    teamId,
                    effectiveDate: (peoStartDate as unknown) as Date,
                    effectivePlainDate: (peoStartDate as unknown) as Date,
                    initialEffectiveDate: (contractStartDate as unknown) as Date,
                    initialEffectivePlainDate: (contractStartDate as unknown) as Date,
                    jobTitleName: jobTitle,
                    data: {
                        ...(contract.data ?? {}),
                        additionalInfo: {
                            ...(contract.data?.additionalInfo ?? {}),
                            payMethod,
                        },
                    },
                },
                {
                    where: {
                        oid: contractOid,
                    },
                    transaction,
                }
            );

            await this.db.UserContract.update(
                {
                    teamId,
                },
                {
                    where: {
                        contractId: contract.id,
                        teamId: contract.teamId,
                    },
                    transaction,
                }
            );

            await this.db.WorkStatement.update(
                {
                    jobTitleName: jobTitle,
                    effectiveDate: (contractStartDate as unknown) as Date,
                    effectivePlainDate: peoStartDate,
                },
                {
                    where: {
                        ContractId: contract.id,
                    },
                    transaction,
                }
            );

            if (peoStartDate) {
                await this.db.HrisProfile.update(
                    {
                        startDate: contractStartDate,
                    },
                    {
                        where: {
                            id: contract.HrisProfileId,
                        },
                        transaction,
                    }
                );
            }

            await this.updatePEOContract({
                deelContractId: contract.id,
                peoStartDate,
                origHireDate: contractStartDate,
                lastHireDate: contractStartDate,
                payRate: String(salary),
                payRateEffectiveDate: effectiveDate,
                jobCode: jobCode || jobTitle,
                jobCodeEffectiveDate: effectiveDate,
                payPeriod: payMethod === PEOPayMethodEnum.HOURLY ? 'H' : 'Y',
                payMethod,
                employeeTypeCode,
                workHoursPerWeek,
                employeeTypeEffectiveDate: effectiveDate,
            });

            if (timeTracking?.scheduleId || timeTracking?.scheduleId === null) {
                let oldScheduleId = null;
                try {
                    const oldTimeTracking = await timeTrackingService.getTimeTrackingContract({
                        oid: contract.oid,
                        contractType: CONTRACT_TYPES.PEO,
                    });
                    oldScheduleId = oldTimeTracking?.schedule?.id ?? null;
                } catch (error) {
                    // do not throw an error here, just log it so that other updates can still proceed
                    this.log.error({
                        message: 'Error fetching old time tracking',
                        error,
                        contractOid: contract.oid,
                    });
                }

                const checkPeoContractWorkSchedulesMissingUpdateStrategy = new CheckPeoContractWorkSchedulesMissingUpdateStrategy();
                if (await checkPeoContractWorkSchedulesMissingUpdateStrategy.isEligible({updateRequestedAt: effectiveDate}, null)) {
                    this.log.info({
                        message: `peoContractService.updateAgreement: sync update eligible: updateRequestedAt ${effectiveDate}, currentDate ${moment()
                            .utc()
                            .format('YYYY-MM-DD')}`,
                    });

                    await timeTrackingService.validateContractScheduleUpdate(contract.oid, CONTRACT_TYPES.PEO, {}, timeTracking.scheduleId);
                } else {
                    this.log.info({
                        message: `peoContractService.updateAgreement: sync update is not eligible: updateRequestedAt ${effectiveDate}, currentDate ${moment()
                            .utc()
                            .format('YYYY-MM-DD')}`,
                    });
                    await peoContractMissingUpdateService.createOrUpdatePeoContractMissingUpdate({
                        deelContractId: contract.id,
                        type: PeoContractMissingUpdateType.WORK_SCHEDULES,
                        updateRequestedAt: effectiveDate,
                        updatedValue: timeTracking.scheduleId,
                        oldValue: oldScheduleId,
                        deelUpdaterProfileId: profile.id,
                        updateReason,
                        additionalInfo: reasonAdditionalInfo,
                    });
                }
            }

            await this.checkCompensationTypeChange(
                {
                    contract,
                    oldPayMethod: contract.peoContract?.payMethod,
                    newPayMethod: payMethod,
                },
                {transaction}
            );

            await employmentService.updateTerms({
                contractOid: contract.oid,
                // @ts-expect-error Incorrect types in updateTerms
                rate: String(salary),
                startDate: (peoStartDate as unknown) as Date,
                effectiveDate: plainEffectiveDate,
                scale: payMethod === PEOPayMethodEnum.HOURLY ? 'HOUR' : 'YEAR',
                clientTeamId: teamId,
                jobTitle,
            });

            const changelogContext = {
                actorEmail: profile.email,
                organizationId: contract.organizationId,
                hrisProfileId: contract.HrisProfileId,
                updateReason,
                contractOid,
                url: asyncContext.originalUrl,
            };

            await this.postSalaryChangeToGPDataChanges({
                profileId: profile.id,
                contractId: contract.id,
                contractOid: contract.oid,
                legalEntityId: contract.clientLegalEntityId,
                salary,
                payMethod,
                peoStartDate,
                effectiveDate: plainEffectiveDate,
                transaction,
                changelogContext,
            });

            if ((scopeId && seniorityId) || jobScope) {
                await peoScopeOfWorkService.upsertPeoScopeOfWork({
                    peoContractId: contract.peoContract.id,
                    ...(seniorityId ? {seniorityId} : {}),
                    ...(scopeId ? {scopeId, jobScope: null} : {}),
                    ...(!scopeId && jobScope ? {jobScope, scopeId: null} : {}),
                });
            }
            const enableModificationReasonCall = featureFlagsProvider.getFlags()?.peo?.enableModificationReasonCall;
            if (enableModificationReasonCall && updateReason) {
                const modificationReasonService = new ModificationReasonService(this.db);
                const fields = [];
                await modificationReasonService.injectFieldModificationReasons({
                    fields,
                    reasonOid: updateReason,
                    additionalInfo: reasonAdditionalInfo,
                    hrisProfileId: contract.HrisProfileId,
                    organizationId: contract.organizationId,
                    transaction,
                });
            }

            // Sidecar
            const payload = {
                contractId: contractOid,
                workStartDate: peoStartDate,
            };
            await this.saveWorkRecordsSidecar({
                payload,
                transaction,
                employmentType: employeeTypeCode,
                workHoursPerWeek,
                seniorityId,
                withContractUpdates: true,
            });
        });

        if (peoStartDate !== oldPeoStartDate) {
            const payload = {
                contractOid,
                previousStartDate: oldPeoStartDate,
                newStartDate: peoStartDate,
            };

            if (isPEOEmploymentEventEnabled()) {
                await peoEmploymentEventService.publishEmploymentUpdated({
                    id: contractOid,
                    startDate: peoStartDate,
                });
            }

            await slackDoAction({type: 'PEO_UPDATE_START_DATE', payload});
        }

        await timeTrackingService.validateContractScheduleAndUnassignIfIncompatible(contract.oid, CONTRACT_TYPES.PEO, {
            employmentType,
            workHoursPerWeek,
            workerType: payMethod === PEOPayMethodEnum.HOURLY ? 'HOURLY_PEO' : 'SALARIED_PEO',
        });
        await timeTrackingService.registerTimeTrackingContractSync({oid: contract.oid, effectiveDate: moment(effectiveDate).toISOString()});
    }

    async updateContractStartDate(
        {
            contractOid,
            forceUpdate,
            deelPeoStartDate: peoStartDateRaw,
            startDate: contractStartDateRaw,
            profileId,
        }: {contractOid?: string; forceUpdate?: boolean; deelPeoStartDate?: string; startDate?: string; profileId?: number} = {forceUpdate: false}
    ) {
        let oldStartDate;
        let newStartDate;

        await this.db.sequelize.transaction(async (transaction) => {
            const contract = await this.getContractByOid({
                deelContractOid: contractOid,
                addPEOData: false,
                additionalAttributes: ['id', 'oid', 'HrisProfileId', 'effectiveDate', 'initialEffectiveDate', 'timezone', 'status'],
            });

            if (!contract) {
                throw new HttpError(404, 'Contract not found');
            }

            const peoStartDate = this._formatDate(peoStartDateRaw, contract.timezone);
            const contractStartDate = this._formatDate(contractStartDateRaw, contract.timezone);
            const oldPeoStartDate = this._formatDate(contract.effectiveDate);
            const oldContractStartDate = this._formatDate(contract.initialEffectiveDate);
            this._validateDatesUpdate(peoStartDate, oldPeoStartDate, contractStartDate, oldContractStartDate);

            newStartDate = peoStartDate;
            oldStartDate = oldPeoStartDate;

            // Check first if employee is already part of payroll
            const employments = await EMSEmploymentsService.searchEmployments({contractOids: [contractOid]});

            if (!employments?.length) {
                throw new HttpError(404, 'Employment not found for given contract');
            }

            const employmentDeelStartDate = await EMSEmploymentsService.getDeelStartDate(employments[0].id);

            if (!employmentDeelStartDate) {
                throw new HttpError(404, 'Employment start date not found for given contract');
            }

            if (employmentDeelStartDate.isLocked && !forceUpdate) {
                throw new HttpError(400, 'Cannot update start date as employee is already part of payroll');
            }

            const isPeoStartDateInFuture = peoStartDate > moment.tz(contract.timezone).format('YYYY-MM-DD');

            if (contract.status === CONTRACT_STATUSES.IN_PROGRESS && isPeoStartDateInFuture) {
                await this.clearPeoContractActiveStatus(contractOid);

                await this.db.Contract.update(
                    {
                        effectiveDate: (peoStartDate as unknown) as Date,
                        effectivePlainDate: (peoStartDate as unknown) as Date,
                        initialEffectiveDate: (contractStartDate as unknown) as Date,
                        initialEffectivePlainDate: (contractStartDate as unknown) as Date,
                        status: CONTRACT_STATUSES.ONBOARDED,
                        activatedAt: null,
                    },
                    {
                        where: {
                            oid: contractOid,
                        },
                        transaction,
                    }
                );

                if (oldStartDate !== newStartDate) {
                    if (isPEOEmploymentEventEnabled()) {
                        await peoEmploymentEventService.publishEmploymentUpdated({
                            id: contractOid,
                            startDate: peoStartDate,
                            status: PEOEmploymentStatus.ONBOARDING,
                        });
                    }
                }
            } else {
                await this.db.Contract.update(
                    {
                        effectiveDate: (peoStartDate as unknown) as Date,
                        effectivePlainDate: (peoStartDate as unknown) as Date,
                        initialEffectiveDate: (contractStartDate as unknown) as Date,
                        initialEffectivePlainDate: (contractStartDate as unknown) as Date,
                    },
                    {
                        where: {
                            oid: contractOid,
                        },
                        transaction,
                    }
                );

                if (oldStartDate !== newStartDate) {
                    if (isPEOEmploymentEventEnabled()) {
                        await peoEmploymentEventService.publishEmploymentUpdated({
                            id: contractOid,
                            startDate: peoStartDate,
                        });
                    }
                }
            }

            await this.db.HrisProfile.update(
                {
                    startDate: contractStartDate,
                },
                {
                    where: {
                        id: contract.HrisProfileId,
                    },
                    transaction,
                }
            );

            await this.db.WorkStatement.update(
                {
                    effectiveDate: (contractStartDate as unknown) as Date,
                    effectivePlainDate: (peoStartDate as unknown) as Date,
                },
                {
                    where: {
                        ContractId: contract.id,
                    },
                    transaction,
                }
            );

            await this.updatePEOContract({
                deelContractId: contract.id,
                peoStartDate,
                origHireDate: contractStartDate,
                lastHireDate: contractStartDate,
            });

            if (!forceUpdate) {
                await EMSEmploymentsService.updateEmployment(employments[0].id, {
                    contractOid,
                    deelStartDate: peoStartDate,
                });
            }

            // @ts-expect-error Incorrect types in updateTerms
            await employmentService.updateTerms({
                contractOid: contract.oid,
                startDate: (peoStartDate as unknown) as Date,
            });

            await this.postSalaryChangeToGPDataChanges({
                profileId,
                contractId: contract.id,
                contractOid: contract.oid,
                legalEntityId: contract.clientLegalEntityId,
                peoStartDate,
            });

            // Sidecar
            const payload = {
                contractId: contract?.oid,
                workStartDate: peoStartDate,
            };
            await this.saveWorkRecordsSidecar({contractId: contract?.id, payload, transaction});
        });

        if (oldStartDate && oldStartDate !== newStartDate) {
            const payload = {
                contractOid,
                previousStartDate: oldStartDate,
                newStartDate,
            };

            await slackDoAction({type: 'PEO_UPDATE_START_DATE', payload});
        }
    }

    async updateContractNotRegular({oid, contractDetails, creator, teamId, organization, shouldSendInvite = true, transaction}) {
        const {
            employmentType,
            jobTitle: jobTitleInput,
            variableCompensations,
            workingLocationId,
            holidays,
            departmentId,
            orgStructures,
            hrisJobPositionId,
            payMethod,
            salary,
            startDate: contractStartDateRaw,
            peoStartDate: peoStartDateRaw,
            workHoursPerWeek,
            zipCode,
            street,
            country,
            city,
            state,
            scopeId,
            jobScope,
            seniorityId,
            payrollSettingsId,
            clientLegalEntityId,
        } = contractDetails;

        let timeOffType = contractDetails.timeOffType;

        const contract = await this.getContractByOid({
            deelContractOid: oid,
            addPEOData: true,
            additionalAttributes: [
                'id',
                'oid',
                'effectiveDate',
                'initialEffectiveDate',
                'country',
                'state',
                'name',
                'timezone',
                'contractorId',
                'HrisProfileId',
                'clientLegalEntityId',
                'invitedClientEmail',
                'data',
                'createdAt',
                'organizationId',
            ],
        });

        const peoStartDate = this._formatDate(peoStartDateRaw, contract.timezone);
        const contractStartDate = this._formatDate(contractStartDateRaw, contract.timezone);
        const oldPeoStartDate = this._formatDate(contract.effectiveDate);
        const oldContractStartDate = this._formatDate(contract.initialEffectiveDate);
        this._validateDatesUpdate(peoStartDate, oldPeoStartDate, contractStartDate, oldContractStartDate);

        const eorCountryData = await this.db.EORCountryData.get({country: 'US', state: contract.state, relatedContractId: contract.id});

        if (timeOffType) {
            timeOffType = holidays !== null && holidays >= 0 ? PEOTimeOffTypeEnum.specific : PEOTimeOffTypeEnum.standard;
        }

        if (employmentType === PEOEmploymentTypeEnum.partTime && timeOffType === PEOTimeOffTypeEnum.standard) {
            timeOffType = this.db.EORCountryData.getPartTimeTimeOffType({eorCountryData, timeOffType});
        }

        const {jobTitle, jobCode} = await this._validateAndGetJobTitleCode(jobTitleInput, contract.clientLegalEntityId);

        const profile = await this.db.Profile.findFirst({
            where: {
                id: contract.contractorId,
                profileType: this.db.Profile.PROFILE_TYPES_ENUM.peo_employee,
            },
            useMaster: false,
            raw: true,
        });

        if (!profile) {
            throw new HttpError(404, 'Profile not found');
        }

        const hrisProfile = await this.db.HrisProfile.findFirst({
            where: {
                id: contract.HrisProfileId,
            },
            transaction,
            useMaster: false,
        });

        await this.db.HrisProfile.update(
            {
                startDate: contractStartDate,
                ...(state ? {state} : {}),
            },
            {
                where: {
                    id: hrisProfile.id,
                },
                transaction,
            }
        );

        const hrisProfileAddress = await this.db.HrisProfileAddresses.findFirst({
            where: {hrisProfileId: contract.HrisProfileId, type: HrisProfileAddressesTypeEnum.HOME},
            attributes: ['id', 'addressId'],
            transaction,
            useMaster: true,
        });

        let employee;
        if (shouldSendInvite) {
            employee = await employmentService.getEmployeeByProfileId(contract.contractorId);
        }

        const address = await this.db.Address.findFirst({
            where: {
                id: profile.AddressId,
            },
            transaction,
            useMaster: false,
        });

        if (!address.geoCode || address.geoCode === '') {
            const matchedAddress = await peoLocationService.validateAndGetEntityGeoCode({
                city,
                geoCode: address.geoCode,
                zipCode,
            });

            const addressSearchOptions = [profile.AddressId, hrisProfileAddress.addressId];
            if (employee && employee.addressId) addressSearchOptions.push(employee.addressId);

            await this.db.Address.update(
                {
                    geoCode: matchedAddress.geoCode,
                    ...(city ? {city} : {}),
                    street,
                    ...(city ? {city} : {}),
                    ...(state ? {state} : {}),
                    ...(country ? {country} : {}),
                },
                {
                    where: {
                        id: {[this.db.Sequelize.Op.in]: addressSearchOptions},
                    },
                    transaction,
                }
            );
        }

        await this.db.Profile.update(
            {
                lastTeam: teamId,
                jobTitle,
            },
            {
                where: {
                    id: profile.id,
                },
                transaction,
            }
        );

        if (departmentId || orgStructures || hrisJobPositionId) {
            // eslint-disable-next-line @typescript-eslint/no-var-requires -- avoid circular dependency
            const {updateHrisProfileFields} = require('../hris/profiles');

            await updateHrisProfileFields({
                updaterProfile: creator,
                profileId: creator.id,
                organizationId: organization.id,
                hrisProfile,
                departmentId,
                orgStructures,
                hrisJobPositionId,
            });
        }

        const workLocation = await peoWorkLocationService.getPEOWorkLocation(workingLocationId);
        const workLocationCode = workLocation.locationId;
        await peoWorkLocationService.validateWorkLocation(workLocation, contract.clientLegalEntityId);

        const isUnderReviewContract = await this._checkIfContractHasPrismResourceRequest(contract.clientLegalEntityId, workLocationCode, jobCode || jobTitle);

        await this.updatePEOContract({
            peoContractStatus: isUnderReviewContract ? PeoContractStatus.UNDERWRITING : PeoContractStatus.STARTED,
            deelContractId: contract.id,
            ...(clientLegalEntityId ? {deelEntityId: String(clientLegalEntityId)} : {}),
            zipCode,
            payRate: String(salary),
            addressLine1: street,
            peoStartDate,
            origHireDate: contractStartDate,
            lastHireDate: contractStartDate,
            ...(employmentType ? {employeeTypeCode: this.mapEmployeeTypeCode(employmentType)} : {}),

            payPeriod: payMethod === PEOPayMethodEnum.HOURLY ? 'H' : 'Y',
            ...(payMethod ? {payMethod} : {payMethod: PEOPayMethodEnum.SALARY}),

            ...(workLocationCode ? {workLocationCode} : {}),

            ...(variableCompensations ? {variableCompensations} : {}),

            ...(jobCode || jobTitle ? {jobCode: jobCode || jobTitle} : {}),
            ...(workHoursPerWeek ? {workHoursPerWeek} : {}),

            ...(city ? {city} : {}),
            ...(state ? {stateCode: state} : {}),
            ...(street ? {addressLine1: street} : {}),
        });

        if (holidays && timeOffType) {
            await peoScopeOfWorkService.upsertPeoScopeOfWork({
                peoContractId: contract.peoContract.id,
                holidays,
                timeOffType,
                ...(scopeId ? {scopeId} : {}),
                ...(!scopeId && jobScope ? {jobScope} : {}),
                ...(seniorityId ? {seniorityId} : {}),
            });
        }

        const prismIcp = await this.db.ICP.findFirst({
            attributes: ['id'],
            where: {
                name: 'PrismHR',
            },
            useMaster: false,
        });

        const employments = await EMSEmploymentsService.searchEmployments({contractOids: [oid], includePayrollSettings: true});

        let employment = null;

        if (employments && employments.length > 0) {
            employment = employments[0];
        }

        if (employment) {
            await employmentService.updateEmployementById(employment.id, {
                contractOid: oid,
                country,
                state,
                clientTeamId: teamId,
                email: contract.invitedClientEmail,
                payrollLegalEntityId: clientLegalEntityId,
                ICPId: prismIcp.id,
                // Will allow us to reuse most of the GP flows for payroll
                gpContractId: contract.id,
            });

            await employmentService.updateTerms({
                contractOid: oid,
                // @ts-expect-error Incorrect types in updateTerms
                rate: String(salary),
                scale: payMethod === PEOPayMethodEnum.HOURLY ? 'HOUR' : 'YEAR',
                effectiveDate: (peoStartDate as unknown) as Date,
            });

            if (employment?.employmentPayrollSettings?.payrollSettingsId && payrollSettingsId !== employment.employmentPayrollSettings.payrollSettingsId) {
                await this.payrollSettingsService.moveGpContractsToPayrollSettings({
                    contractOids: [oid],
                    targetPayrollSettingsId: payrollSettingsId,
                    effectiveDate: moment().format('YYYY-MM-DD'),
                    req: {},
                });
            }
        } else {
            await employmentService.createPEOEmployment({
                contractOid: oid,
                country,
                state,
                clientTeamId: teamId,
                // @ts-expect-error Incorrect types in createPEOEmployment
                rate: String(salary),
                scale: payMethod === PEOPayMethodEnum.HOURLY ? 'HOUR' : 'YEAR',
                email: contract.invitedClientEmail,
                // @ts-expect-error Incorrect types in createPEOEmployment
                startDate: peoStartDate,
                employeeKey: employee.id,
                employeeId: profile.id,
                jobTitle,

                ...(clientLegalEntityId ? {payrollLegalEntityId: clientLegalEntityId} : {}),
                ICPId: prismIcp.id,
                // Will allow us to reuse most of the GP flows for payroll
                gpContractId: contract.id,
                payrollSettingsId,
            });
        }

        await this.postSalaryChangeToGPDataChanges({
            profileId: creator.id,
            contractId: contract.id,
            contractOid: contract.oid,
            legalEntityId: contract.clientLegalEntityId,
            salary,
            payMethod: payMethod ?? PEOPayMethodEnum.SALARY,
            effectiveDate: peoStartDate,
        });

        const dataToUpdate = {
            teamId,
            ...(jobTitle ? {name: `${contract.name.split('-')[0].trim()} - ${jobTitle}`} : {}),
            country,
            state,
            effectiveDate: (peoStartDate as unknown) as Date,
            effectivePlainDate: (peoStartDate as unknown) as Date,
            initialEffectiveDate: (peoStartDate as unknown) as Date,
            initialEffectivePlainDate: (peoStartDate as unknown) as Date,
            jobTitleName: jobTitle,
            status: profile.userId ? CONTRACT_STATUSES.ONBOARDING : CONTRACT_STATUSES.WAITING_FOR_EMPLOYEE_SIGN,
            data: {
                ...(contract.data ?? {}),
                additionalInfo: {
                    ...(contract.data?.additionalInfo ?? {}),
                    payMethod,
                },
            },
            ...(clientLegalEntityId ? {clientLegalEntityId} : {}),
        };

        await this.db.Contract.update(dataToUpdate, {
            where: {
                id: contract.id,
            },
            transaction,
        });

        await this.db.UserContract.update(
            {
                teamId,
            },
            {
                where: {
                    contractId: contract.id,
                },
                transaction,
            }
        );

        // Sidecar
        const payload = {
            contractId: contract?.oid,
            workStartDate: peoStartDate,
        };
        await this.saveWorkRecordsSidecar({
            contractId: contract?.id,
            payload,
            transaction,
            employmentType,
            workHoursPerWeek,
            seniorityId,
            withContractUpdates: true,
        });

        await this.checkCompensationTypeChange(
            {
                contract,
                oldPayMethod: contract.peoContract?.payMethod,
                newPayMethod: payMethod,
            },
            {transaction}
        );

        if (oldPeoStartDate !== peoStartDate) {
            if (isPEOEmploymentEventEnabled()) {
                await peoEmploymentEventService.publishEmploymentUpdated({
                    id: contract.oid,
                    startDate: peoStartDate,
                    status: PEOEmploymentStatus.ONBOARDING,
                });
            }
        }

        const hrisProfileData = {
            firstName: profile.firstName,
            lastName: profile.lastName,
            email: contract.invitedClientEmail,
            employeeNationality: contract.country,
        };

        transaction.afterCommit(async () => {
            if (shouldSendInvite) {
                await this.sendPEOContractInvitation(
                    {...hrisProfileData, profileId: profile.id, userId: hrisProfile.userId},
                    organization,
                    {oid, ...contract, ...dataToUpdate},
                    address
                );
            }

            await timeTrackingService.validateContractScheduleUpdate(
                oid,
                CONTRACT_TYPES.PEO,
                {
                    employmentType,
                    workHoursPerWeek,
                    workerType: payMethod === PEOPayMethodEnum.HOURLY ? 'HOURLY_PEO' : 'SALARIED_PEO',
                },
                contractDetails.timeTracking?.scheduleId
            );
        });

        return contract;
    }

    async updateContractPartially({oid, contractDetails, creator, teamId, organization}) {
        const {
            employmentType,
            jobTitle: jobTitleInput,
            variableCompensations,
            workingLocationId,
            holidays,
            departmentId,
            orgStructures,
            hrisJobPositionId,
            payMethod,
            salary,
            startDate: contractStartDateRaw,
            peoStartDate: peoStartDateRaw,
            workHoursPerWeek,
            zipCode,
            street,
            country,
            county,
            city,
            state,
            jobScope,
            scopeId,
            seniorityId,
        } = contractDetails;

        let timeOffType = contractDetails.timeOffType;

        const contract = await this.getContractByOid({
            deelContractOid: oid,
            addPEOData: true,
            additionalAttributes: [
                'id',
                'effectiveDate',
                'initialEffectiveDate',
                'country',
                'state',
                'name',
                'contractorId',
                'HrisProfileId',
                'clientLegalEntityId',
                'invitedClientEmail',
                'teamId',
                'data',
                'createdAt',
                'organizationId',
            ],
        });

        this.log.info({message: 'peoContract', peoContract: contract?.peoContract});

        const timezone = creator?.timezone || 'America/Los_Angeles';
        const peoStartDate = this._formatDate(peoStartDateRaw, timezone);
        const contractStartDate = this._formatDate(contractStartDateRaw, timezone);
        const oldPeoStartDate = this._formatDate(contract.effectiveDate);
        const oldContractStartDate = this._formatDate(contract.initialEffectiveDate);

        const eorCountryData = await this.db.EORCountryData.get({country: 'US', state: contract.state, relatedContractId: contract.id});

        if (timeOffType) {
            timeOffType = holidays !== null && holidays >= 0 ? PEOTimeOffTypeEnum.specific : PEOTimeOffTypeEnum.standard;
        }

        if (employmentType === PEOEmploymentTypeEnum.partTime && timeOffType === PEOTimeOffTypeEnum.standard) {
            timeOffType = this.db.EORCountryData.getPartTimeTimeOffType({eorCountryData, timeOffType});
        }

        if (
            contract?.peoContract?.peoContractStatus === PeoContractStatus.ACTIVE ||
            contract?.peoContract?.peoContractStatus === PeoContractStatus.IN_PROGRESS
        ) {
            this._validateDatesUpdate(peoStartDate, oldPeoStartDate, contractStartDate, oldContractStartDate);
        }

        const {jobTitle, jobCode} = await this._validateAndGetJobTitleCode(jobTitleInput, contract.clientLegalEntityId);

        const jobTitleExists = jobTitle !== undefined && jobTitle !== null;

        let workLocationCode;
        let isUnderReviewContract;

        if (workingLocationId) {
            const workLocation = await peoWorkLocationService.getPEOWorkLocation(workingLocationId);
            workLocationCode = workLocation.locationId;
            await peoWorkLocationService.validateWorkLocation(workLocation, contract.clientLegalEntityId);
            isUnderReviewContract = await this._checkIfContractHasPrismResourceRequest(contract.clientLegalEntityId, workLocationCode, jobCode || jobTitle);
        }

        const profile = await this.db.Profile.findFirst({
            where: {
                id: contract.contractorId,
                profileType: this.db.Profile.PROFILE_TYPES_ENUM.peo_employee,
            },
            useMaster: false,
            raw: true,
        });

        if (!profile) {
            throw new HttpError(404, 'Profile not found');
        }

        const address = await this.db.Address.findFirst({
            where: {
                id: profile.AddressId,
            },
            useMaster: false,
        });

        let matchedAddress;

        if (!address.geoCode || address.geoCode === '') {
            if (city && zipCode) {
                matchedAddress = await peoLocationService.validateAndGetEntityGeoCode({
                    city,
                    geoCode: address.geoCode ?? null,
                    zipCode,
                });
            }
        }

        await this.db.sequelize.transaction(async (transaction) => {
            const hrisProfileAddress = await this.db.HrisProfileAddresses.findFirst({
                where: {hrisProfileId: contract.HrisProfileId, type: HrisProfileAddressesTypeEnum.HOME},
                attributes: ['id', 'addressId'],
                transaction,
                useMaster: true,
            });

            await this.db.UserContract.update(
                {
                    teamId,
                },
                {
                    where: {
                        contractId: contract.id,
                    },
                    transaction,
                }
            );

            await this.db.HrisProfile.update(
                {
                    ...(contractStartDate ? {startDate: contractStartDate} : {}),
                    ...(state ? {state} : {}),
                },
                {
                    where: {
                        id: contract.HrisProfileId,
                    },
                    transaction,
                }
            );

            const addressSearchOptions = [];
            if (hrisProfileAddress && hrisProfileAddress.addressId) addressSearchOptions.push(hrisProfileAddress.addressId);
            if (profile && profile.AddressId) addressSearchOptions.push(profile.AddressId);

            if (addressSearchOptions.length > 0) {
                await this.db.Address.update(
                    {
                        ...(matchedAddress?.geoCode ? {geoCode: matchedAddress.geoCode} : {}),
                        ...(zipCode ? {zip: zipCode} : {}),
                        ...(street ? {street} : {}),
                        ...(city ? {city} : {}),
                        ...(state ? {state} : {}),
                        ...(country ? {country} : {}),
                    },
                    {
                        where: {
                            id: {[this.db.Sequelize.Op.in]: addressSearchOptions},
                        },
                        transaction,
                    }
                );
            }

            await this.db.Profile.update(
                {
                    ...(teamId ? {lastTeam: teamId} : {}),
                    ...(jobTitle ? {jobTitle} : {}),
                },
                {
                    where: {
                        id: profile.id,
                    },
                    transaction,
                }
            );

            if (departmentId || orgStructures || hrisJobPositionId) {
                // eslint-disable-next-line @typescript-eslint/no-var-requires -- avoid circular dependency
                const {updateHrisProfileFields} = require('../hris/profiles');

                const hrisProfile = await this.db.HrisProfile.findFirst({
                    where: {
                        id: contract.HrisProfileId,
                    },
                    transaction,
                    useMaster: false,
                });

                await updateHrisProfileFields({
                    updaterProfile: creator,
                    profileId: creator.id,
                    organizationId: organization.id,
                    hrisJobPositionId,
                    hrisProfile,
                    departmentId,
                    orgStructures,
                    transaction,
                });
            }

            await this.db.Contract.update(
                {
                    ...(teamId ? {teamId} : {}),
                    ...(jobTitleExists ? {name: `${contract.name.split('-')[0].trim()} - ${jobTitle}`} : {name: contract.name}),
                    ...(country ? {country} : {}),
                    ...(state ? {state} : {}),

                    ...(peoStartDate ? {effectiveDate: (peoStartDate as unknown) as Date} : {}),
                    ...(peoStartDate ? {effectivePlainDate: (peoStartDate as unknown) as Date} : {}),
                    ...(contractStartDate ? {initialEffectiveDate: (contractStartDate as unknown) as Date} : {}),
                    ...(contractStartDate ? {initialEffectivePlainDate: (contractStartDate as unknown) as Date} : {}),
                    ...(jobTitleExists ? {jobTitleName: jobTitle} : {}),
                    ...(payMethod && {
                        data: {
                            ...(contract.data ?? {}),
                            additionalInfo: {
                                ...(contract.data?.additionalInfo ?? {}),
                                payMethod,
                            },
                        },
                    }),
                    status: CONTRACT_STATUSES.UNDER_REVIEW,
                },
                {
                    where: {
                        id: contract.id,
                    },
                    transaction,
                }
            );

            // Sidecar
            await this.saveWorkRecordsSidecar({
                contractId: contract.id,
                payload: {
                    contractId: oid,
                    workStartDate: peoStartDate,
                },
                transaction,
                employmentType,
                workHoursPerWeek,
                seniorityId,
                withContractUpdates: true,
            });

            transaction.afterCommit(async () => {
                await this.updatePEOContract({
                    peoContractStatus: isUnderReviewContract ? PeoContractStatus.UNDERWRITING : PeoContractStatus.STARTED,
                    deelContractId: contract.id,

                    ...(zipCode ? {zipCode} : {}),
                    ...(salary ? {payRate: String(salary)} : {}),
                    peoStartDate,
                    origHireDate: contractStartDate,
                    lastHireDate: contractStartDate,
                    ...(employmentType ? {employeeTypeCode: this.mapEmployeeTypeCode(employmentType)} : {}),
                    payPeriod: payMethod === PEOPayMethodEnum.HOURLY ? 'H' : 'Y',
                    ...(payMethod ? {payMethod} : {payMethod: PEOPayMethodEnum.SALARY}),

                    ...(workLocationCode ? {workLocationCode} : {}),

                    ...(variableCompensations ? {variableCompensations} : {}),

                    ...(jobCode || jobTitle ? {jobCode: jobCode || jobTitle} : {}),
                    ...(workHoursPerWeek ? {workHoursPerWeek} : {}),

                    ...(city ? {city} : {}),
                    ...(state ? {stateCode: state} : {}),
                    ...(street ? {addressLine1: street} : {}),
                    ...(county ? {county} : {}),
                });

                if (holidays || timeOffType || scopeId || jobScope || seniorityId) {
                    await peoScopeOfWorkService.upsertPeoScopeOfWork({
                        peoContractId: contract.peoContract.id,
                        ...(holidays ? {holidays} : {}),
                        ...(timeOffType ? {timeOffType} : {}),
                        ...(scopeId ? {scopeId, jobScope: null} : {}),
                        ...(!scopeId && jobScope ? {jobScope, scopeId: null} : {}),
                        ...(seniorityId ? {seniorityId} : {}),
                    });
                }

                if (salary && payMethod) {
                    await this.postSalaryChangeToGPDataChanges({
                        profileId: creator.id,
                        contractId: contract.id,
                        contractOid: contract.oid,
                        legalEntityId: contract.clientLegalEntityId,
                        salary,
                        payMethod,
                        peoStartDate,
                    });
                }

                await this.checkCompensationTypeChange({
                    contract,
                    oldPayMethod: contract.peoContract?.payMethod,
                    newPayMethod: payMethod,
                });

                await timeTrackingService.validateContractScheduleUpdate(
                    oid,
                    CONTRACT_TYPES.PEO,
                    {
                        employmentType,
                        workHoursPerWeek,
                        workerType: payMethod === PEOPayMethodEnum.HOURLY ? 'HOURLY_PEO' : 'SALARIED_PEO',
                    },
                    contractDetails.timeTracking?.scheduleId
                );
            });
        });

        return {contract, peoContract: contract?.peoContract};
    }

    async deleteDraftContractNotRegular(oid) {
        await this.db.sequelize.transaction(async (transaction) => {
            try {
                const contract = await this.getContractByOid({
                    deelContractOid: oid,
                    addPEOData: true,
                    additionalAttributes: ['id', 'contractorId', 'HrisProfileId', 'status'],
                });

                if (!contract || contract.peoContract.origin === PeoContractOrigin.Regular || contract.status !== CONTRACT_STATUSES.UNDER_REVIEW) {
                    throw new HttpError(400, `Contract ${oid} cannot be deleted`);
                }

                await this.db.WorkStatement.destroy({
                    where: {
                        ContractId: contract.id,
                    },
                    transaction,
                });

                await this.db.HrisProfileStatic.destroy({
                    where: {
                        primary_contract_id: contract.id,
                    },
                    transaction,
                });

                await this.db.UserContract.destroy({
                    where: {
                        contractId: contract.id,
                    },
                    transaction,
                });

                await this.db.EmployeeOnboardingStep.destroy({
                    where: {
                        contractId: contract.id,
                    },
                    transaction,
                });

                await this.db.ContractActionEvent.destroy({
                    where: {
                        contractId: contract.id,
                    },
                    transaction,
                });

                await this.db.HrisEmployments.destroy({
                    where: {
                        contractId: contract.id,
                    },
                    transaction,
                });

                if (contract?.peoContract?.id) {
                    await this.deletePEOContract(contract.peoContract.id);
                }

                await this.db.Contract.destroy({
                    where: {
                        oid,
                    },
                    transaction,
                });
            } catch (error) {
                this.log.error({message: 'Failed to delete draft contract not regular', error});
                throw error;
            }
        });
    }

    async updateCompensationOnEffectiveDate({contractOid, payload, profile}) {
        const {
            employmentType,
            compensationEffectiveDate,
            reason: updateReason,
            additionalDetail: reasonAdditionalInfo,
            payRate,
            payMethod,
            workHoursPerWeek,
        } = payload;

                'teamId',

                'oid',
                'HrisProfileId',

                'clientLegalEntityId',

                'contractorId',
                'createdAt',
                'organizationId',
            ],
        });

        const plainCompensationEffectiveDate = moment(compensationEffectiveDate).tz(contract.timezone).format('YYYY-MM-DD');

        const employeeTypeCode = this.mapEmployeeTypeCode(employmentType);

            workHoursPerWeek,
            contract,
            compensationEffectiveDate,
            payRate,
            payMethod,
            profile,

            reasonAdditionalInfo,
            plainCompensationEffectiveDate,

        });

                    ...(contract?.data ?? {}),
                    additionalInfo: {
                        ...(contract?.data?.additionalInfo ?? {}),
                        payMethod,
                    },
                },
            },
            {
                where: {
                    oid: contractOid,
                },
            }
        );

        await this.updatePEOContract({
            deelContractId: contract.id,
            payRate: String(payRate),
            payPeriod: payMethod === PEOPayMethodEnum.HOURLY ? 'H' : 'Y',
            payMethod,
            employeeTypeCode,
            workHoursPerWeek,
            payRateEffectiveDate: compensationEffectiveDate,
            employeeTypeEffectiveDate: compensationEffectiveDate,
        });

        await this.checkCompensationTypeChange({
            contract,
            oldPayMethod: contract.peoContract?.payMethod,
            newPayMethod: payMethod,
        });

        await employmentService.updateTerms({
            contractOid: contract.oid,
            // @ts-expect-error Incorrect types in updateTerms
            rate: String(payRate),
            scale: payMethod === PEOPayMethodEnum.HOURLY ? 'HOUR' : 'YEAR',

        });

        await this.postSalaryChangeToGPDataChanges({
            profileId: profile.id,
            contractId: contract.id,
            contractOid: contract.oid,
            legalEntityId: contract.clientLegalEntityId,
            salary: payRate,
            payMethod,
            effectiveDate: plainCompensationEffectiveDate,
        });

        await timeTrackingService.registerTimeTrackingContractSync({oid: contract.oid, effectiveDate: moment(compensationEffectiveDate).toISOString()});
    }

    _formatDate(date, timezone = undefined) {
        if (!date) {
            return null;
        }

        if (!timezone) {
            return moment(date).format('YYYY-MM-DD');
        }

        return moment.tz(date, timezone).format('YYYY-MM-DD');
    }

    _validateDatesUpdate(newPeoStartDate, oldPeoStartDate, newContractStartDate, oldContractStartDate) {
        if (!newContractStartDate || oldContractStartDate === newContractStartDate) {
            return;
        }

        if (!newPeoStartDate) {
            throw new HttpError(400, '"startDate" validation failed: update requires "peoStartDate" update');
        }
    }

                },
                {
                    where: {id: contractId},
                    transaction,
                }
            );
        }
    }

    async upsertPEOContractFromProfile({profileId, payload, updateUserProfile = false, transaction}) {
        const contract = await this.getDefaultContract(profileId, ['id', 'clientLegalEntityId', 'contractorId', 'jobTitleName']);

        if (!contract) {
            this.log.error({message: 'no PEO contract was found for that user', profileId});
            throw new ResourceNotFoundError({externalMessage: 'No PEO contract was found'});
        }

        await this.db.DraftEmployeeInfo.destroy({
            where: {
                contractId: contract.id,
                employeeId: profileId,
            },
            transaction,
            force: true,
        });

        const personal = payload.company?.personal;
        const formattedDob = moment(payload.dob).format('YYYY-MM-DD');

        const dataToUpdate = {
            deelEntityId: String(contract.clientLegalEntityId),
            deelContractId: contract.id,

            ...(payload.firstName ? {firstName: payload.firstName} : {}),
            ...(payload.lastName ? {lastName: payload.lastName} : {}),
            ...(payload.dob ? {birthDate: formattedDob} : {}),
            ...(personal?.ssn ? {ssn: personal?.ssn} : {}),

            ...(payload.peo?.gender ? {gender: payload.peo.gender} : {}),
            ...(payload.peo?.maritalStatus ? {maritalStatus: payload.peo.maritalStatus} : {}),
            // Fields are going to be retrieved from a form W4 and I9
            ...(payload.peo?.citizenStatus ? {citizenStatus: payload.peo.citizenStatus} : {}),
            ...(payload.peo?.fedFileStatus ? {fedFileStatus: payload.peo.fedFileStatus} : {}),
            // Update profile to contract
            ...(personal?.street ? {addressLine1: personal.street} : {}),
            ...(personal?.city ? {city: personal.city} : {}),
            ...(personal?.state ? {stateCode: personal.state} : {}),
            ...(personal?.zip ? {zipCode: personal.zip} : {}),

            ...(payload.county ? {county: payload.county} : {}),
        };

        if (Object.entries(dataToUpdate).length <= 2) {
            // If just two new fields to update are the ids so no new properties to update
            return;
        }

        const updateContract = {
            ...(personal?.state ? {state: personal.state} : {}),
        };

        if (Object.entries(updateContract).length > 0) {
            await this.db.Contract.update(updateContract, {where: {id: contract.id}, transaction});
        }

        if (updateUserProfile) {
            await this.updateUserProfile(profileId, payload, transaction);
        }

        const updatedContract = await this.updatePEOContract(dataToUpdate);

        const employeeData = {
            profileId: contract.contractorId,
            ...(payload.firstName ? {firstName: payload.firstName} : {}),
            ...(payload.lastName ? {lastName: payload.lastName} : {}),
            ...(payload.middleName ? {middleName: payload.middleName} : {}),
            ...(payload.preferredName ? {preferredName: payload.preferredName} : {}),
            ...(payload.dob ? {birthDate: formattedDob} : {}),
            ...(payload.nationality ? {nationality: payload.nationality} : {}),
            ...(payload.maritalStatus ? {maritalStatus: payload.maritalStatus} : {}),
            ...(payload.homePhone ? {homePhone: payload.homePhone} : {}),
            ...(payload.mobilePhone ? {mobilePhone: payload.mobilePhone} : {}),
        };

        await this.upsertEmployeeDataForPayroll(employeeData);

        await this._updateContractNameIfNeeded(
            contract.id,
            {
                ...payload,
                jobTitle: contract.jobTitleName,
            },
            transaction
        );

        if (updatedContract.hasAdditionalInfo) {
            await this.db.EmployeeOnboardingStep.update(
                {
                    // @ts-expect-error Incorrect types in EmployeeOnboardingStep
                    status: this.db.EmployeeOnboardingStep.STATUSES.DONE,
                    endedAt: moment.utc().toDate(),
                },
                {
                    where: {
                        contractId: contract.id,
                        name: PEO_EMPLOYEE_CHECKLIST_STEP_TYPE.SUBMIT_ADDITIONAL_DETAILS,
                    },
                    transaction,
                    individualHooks: true,
                }
            );
            await OnboardingPublishersHooks(this.db.Sequelize, this.db.sequelize).onEvaluateProfiles([profileId], transaction);
        }

        return updatedContract;
    }

    async cancelPEOContract({contractOid, reason, transaction = null}) {
        // eslint-disable-next-line @typescript-eslint/no-var-requires -- avoid circular dependency
        const {cancelTransition} = require('../../modules/transitions/services/transition/cancel');
        const peoTransactionalEventService = new PeoTransactionalEventService(this.db);

        await this.db.sequelize.transaction({transaction}, async (transaction) => {
            const enabledAnyContractCancellation = featureFlagsProvider.getFlags()?.frontendFlags?.peoAllContractCancellationEnabled;

            const contract = await this.db.Contract.findFirst({
                attributes: ['id', 'oid', 'status', 'note', 'timezone'],
                where: {
                    oid: contractOid,
                },
                useMaster: Boolean(transaction),
                transaction,
            });

            if (!contract) {
                throw new HttpError.NotFound('Contract not found');
            }

            if (this.DEACTIVATED_CONTRACT_STATUSES.includes(contract.status)) {
                throw new HttpError.BadRequest('Contract already cancelled');
            }

            const isCancellationAllowed = await this.isCancellationAllowed({deelContractId: contract.id});
            if (!enabledAnyContractCancellation && !isCancellationAllowed) {
                throw new HttpError.BadRequest('Contract is active and cannot be cancelled');
            }

            const completionDate = Date.now();
            await contract.update(
                {
                    status: CONTRACT_STATUSES.CANCELLED,
                    completionDate,
                    note: reason && contract.note ? `${reason}\n${contract.note}` : reason,
                    completedAt: completionDate,
                    cancelledAt: completionDate,
                    completionPlainDate: utils.toPlainDate(completionDate, contract.timezone),
                } as unknown,
                {
                    transaction,
                }
            );
            this.log.info({message: `contract ${contractOid} cancelled on ${Date.now()} by cancelPEOContract`});

            await peoTransactionalEventService.createAndRelayEvent({
                eventType: PeoTransactionOutboxEventTypes.CONTRACT_TERMINATION,
                data: {
                    contractOid,
                },
                transaction,
            });

            //todo: use Transactional Outbox to cancel employment ?
            await this.cancelEmploymentOnContractDeactivation(contractOid, completionDate);

            await cancelTransition({newContractOid: contractOid, transaction});

            // Sidecar
            const payload = {
                contractId: contractOid,
                workStatus: WorkStatusEnum.TERMINATED,
                workCompletionDate: completionDate,
            };
            await this.saveWorkRecordsSidecar({payload, transaction});

            if (isPEOEmploymentEventEnabled()) {
                await peoEmploymentEventService.publishEmploymentUpdated({
                    id: contract.oid,
                    endDate: contract.timezone
                        ? moment(completionDate).tz(contract.timezone).format('YYYY-MM-DD')
                        : moment(completionDate).format('YYYY-MM-DD'),
                    status: PEOEmploymentStatus.ENDED,
                });
            }
        });
    }

    async deletePEOContractWithReason({contractOid, reason}) {
        const peoTransactionalEventService = new PeoTransactionalEventService(this.db);
        await this.db.sequelize.transaction(async (transaction) => {
            const enabledAnyContractCancellation = featureFlagsProvider.getFlags()?.frontendFlags?.peoAllContractCancellationEnabled;
            const contract = await this.db.Contract.findFirst({
                attributes: ['id', 'status', 'note', 'deletedAt', 'timezone', 'oid'],
                where: {
                    oid: contractOid,
                },
                useMaster: Boolean(transaction),
                transaction,
                lock: {
                    of: this.db.Contract,
                    level: transaction.LOCK.NO_KEY_UPDATE,
                },
            });

            if (!contract) {
                throw new HttpError.NotFound('Contract not found');
            }
            if (contract.deletedAt) {
                throw new HttpError.NotFound('Contract already deleted');
            }

            if (!this.DEACTIVATED_CONTRACT_STATUSES.includes(contract.status)) {
                throw new HttpError.BadRequest('Contract is active and cannot be deleted. Please cancel the contract first');
            }

            const isCancellationAllowed = await this.isCancellationAllowed({deelContractId: contract.id});
            if (!enabledAnyContractCancellation && !isCancellationAllowed) {
                throw new HttpError.BadRequest('Contract is active and cannot be deleted');
            }

            const completionDate = Date.now();
            if (reason) {
                const updatedNote = contract.note ? `${reason}\n${contract.note}` : reason;

                await contract.update(
                    {
                        status: CONTRACT_STATUSES.CANCELLED,
                        completionDate,
                        completedAt: completionDate,
                        cancelledAt: completionDate,
                        completionPlainDate: utils.toPlainDate(completionDate, contract.timezone),
                        note: updatedNote,
                    } as unknown,
                    {
                        transaction,
                    }
                );
                this.log.info({message: `contract ${contractOid} cancelled on ${Date.now()} by deletePEOContractWithReason`});
            }

            await peoTransactionalEventService.createAndRelayEvent({
                eventType: PeoTransactionOutboxEventTypes.CONTRACT_TERMINATION,
                data: {
                    contractOid,
                },
                transaction,
            });

            if (isPEOEmploymentEventEnabled()) {
                await peoEmploymentEventService.publishEmploymentUpdated({
                    id: contract.oid,
                    endDate: contract.timezone
                        ? moment(completionDate).tz(contract.timezone).format('YYYY-MM-DD')
                        : moment(completionDate).format('YYYY-MM-DD'),
                    status: PEOEmploymentStatus.ENDED,
                });
            }

            //todo: Use Transactional Outbox to cancel employment and other staff
            await this.cancelEmploymentOnContractDeactivation(contractOid, completionDate);

            await contract.destroy({transaction});
        });
    }

    async enrichPEOContractWithHrisProfileInfo(
        peoContract: PeoContract,
        includeAttributes: string[] = ['firstName', 'lastName', 'startDate']
    ): Promise<PeoContract> {
        try {
            if (!peoContract) {
                return peoContract;
            }
            const {deelContractOid, firstName, lastName, origHireDate, lastHireDate} = peoContract;
            this.log.info({peoContract}, 'Enriching PeoContract with HRIS profile data');
            const hrisProfileInfo = await this.db.HrisProfile.findFirst({
                attributes: includeAttributes,
                include: [
                    {
                        model: this.db.Contract,
                        attributes: [],
                        where: {
                            oid: deelContractOid,
                        },
                        required: true,
                    },
                ],
                useMaster: false,
                raw: true,
            });

            if (hrisProfileInfo) {
                this.log.info({hrisProfileInfo}, 'hrisProfileInfo');
                peoContract.firstName = hrisProfileInfo?.firstName ?? firstName;
                peoContract.lastName = hrisProfileInfo?.lastName ?? lastName;
                peoContract.origHireDate = hrisProfileInfo?.startDate ?? origHireDate;
                peoContract.lastHireDate = hrisProfileInfo?.startDate ?? lastHireDate;
            }
            this.log.info({message: 'HRIS profile date enriched', peoContract});
            return peoContract;
        } catch (error) {
            this.log.error({message: 'Error enriching PEO contract with hris profile info', error});
            return peoContract;
        }
    }

    async isCancellationAllowed({deelContractId, deelContractOid}: {deelContractId?: number; deelContractOid?: string}) {

    async getPEOContractsByFilter(filter, {includePayrollSettings = false, includeActiveSalary = false} = {}) {
        const {data: peoContracts, status} = await this.client.get(PEOEndpoints.getPeoContractsByFilter, {
            params: filter,
        });

        if (status === 404) {
            return [];
        }

        if (status !== 200) {
            this.log.error({message: 'Integration to get peo contracts by filter failed', peoContracts});
            throw new HttpError(status, `Integration to get peo contracts by filter failed: ${peoContracts.message}`);
        }
        const instance = PmsEmsSelectionService.selectServiceInstance({legalEntityId: 'peo_organizations', models: ['Groups']});
        if (includePayrollSettings || includeActiveSalary) {
            const payrollSettingsMap = {};
            for (const peoContract of peoContracts) {
                const payrollSettingsIds = await instance.getPayrollSettingsForContractOid(peoContract.deelContractOid);
                if (!payrollSettingsIds?.length) {
                    peoContract.payrollSettings = null;
                    continue;
                }
                const payrollSettingId = payrollSettingsIds[0]?.id;
                if (!payrollSettingsMap[payrollSettingId]) {
                    const payrollSettings = await instance.getById(payrollSettingId);
                    payrollSettingsMap[payrollSettingId] = payrollSettings;
                }
                peoContract.payrollSettings = payrollSettingsMap[payrollSettingId];
            }
        }

        if (includeActiveSalary) {
            // Fetch active salary data changes for all contracts at once
            const contractOids = peoContracts.filter((contract) => contract.payrollSettings).map((contract) => contract.deelContractOid);

            // eslint-disable-next-line @typescript-eslint/no-var-requires -- avoid circular dependency
            const {gpContractDataChangesService} = require('../globalPayroll/contract_data_changes/gp_contract_data_changes_service');
            const activeSalaryChanges = contractOids.length > 0 ? await gpContractDataChangesService.Salary.getActiveDataForContract({contractOids}) : [];

            // Create a map for quick lookup
            const salaryChangesByContractOid = new Map(activeSalaryChanges.map((change) => [change.contractOid, change]));

            for (const peoContract of peoContracts) {
                peoContract.currentCompensationDetails = {rate: null, scale: null, effectiveDate: null, workHoursPerWeek: null};

                const activeSalaryChange = peoContract.payrollSettings ? salaryChangesByContractOid.get(peoContract.deelContractOid) : null;

                if (!activeSalaryChange) {
                    continue;
                }

                try {
                    const salaryData = activeSalaryChange as {data: {rate: number; scale: string}; effectiveDate: string};
                    peoContract.currentCompensationDetails = {
                        rate: salaryData.data?.rate ?? null,
                        scale: salaryData.data?.scale ?? null,
                        effectiveDate: salaryData.effectiveDate ?? null,
                        workHoursPerWeek: peoContract.workHoursPerWeek,
                    };
                } catch (e) {
                    this.log.error({
                        message: 'Error processing salary data change to determine current salary details',
                        err: e.message,
                        deelContractOid: peoContract.deelContractOid,
                    });
                }
            }
        }

        return peoContracts;
    }

    async getPEOContractDetailsByOid({
        deelContractOid,
        getWorkLocationState = false,
        getWithPosition = false,
        getAddress = false,
        getPayDate = false,
        getPayrollFunding = false,
        getEmployment = false,
        getEntitlement = false,
        getContract = false,
    }) {
        const data = await this.getContractDetailsFromPEOService(deelContractOid, getWorkLocationState, getWithPosition);
        const isCancellationAllowed = await this.isCancellationAllowed({deelContractOid});
        const address = getAddress
            ? await this._getAddressFromContractOid({deelContractOid, attributes: ['zip', 'street', 'city', 'state', 'country', 'province']})
            : null;
        const payrollCalendar = getPayDate ? await this.getUpcomingPayrollCalendar(deelContractOid) : null;
        const payrollFunding = getPayrollFunding ? await this.getPayrollFunding(deelContractOid) : null;
        const employment = getEmployment ? await this.getEmploymentTerms(deelContractOid) : null;
        const entitlement = getEntitlement ? await this.getEntitlement(deelContractOid) : null;

        if (typeof data !== 'object' || data === null) {
            this.log.warn({message: `Returned data is not an object, data: ${data}`});
            return data;
        }

        if (typeof data.peoContract === 'object' && data.peoContract !== null) {
            if (getAddress && address) {
                data.peoContract.address = address;
            }

            if (getPayDate && payrollCalendar?.payDate) {
                data.peoContract.payDate = payrollCalendar.payDate;
            }

            if (getPayrollFunding && payrollFunding) {
                data.peoContract.payrollFunding = payrollFunding;
            }

            if (getEmployment && employment) {
                data.peoContract.employment = employment;
            }

            if (getEntitlement && entitlement) {
                data.peoContract.entitlement = {
                    accrued: (entitlement as {accrued: string}).accrued ?? '0',
                };
            }

            if (getContract) {
                const orgId = data.peoContract.organizationId;
                const useNewEligibilityCheck = peoLegalEntityService.shouldUseNewEligibilityCheck(orgId);
                const contract = await this.db.Contract.findFirst({
                    attributes: ['oid', 'status', 'name', 'teamId', 'country'],
                    include: [
                        {
                            attributes: ['id', 'publicId', 'name'],
                            model: this.db.LegalEntity,
                            as: 'ClientLegalEntity',
                            required: true,
                            where: {
                                // TODO: Remove isPeo legacy filter when new PEO eligibility check is fully deployed (PEOP-562)
                                ...(useNewEligibilityCheck ? {} : {isPeo: true}),
                            },
                            include: [...(useNewEligibilityCheck ? [peoLegalEntityService.getPEOProductCodeInclude()] : [])],
                        },
                        {
                            model: this.db.Profile,
                            as: 'Contractor',
                            attributes: ['firstName', 'lastName', 'email'],
                            required: true,
                        },
                    ],
                    where: {
                        contractType: CONTRACT_TYPES.PEO,
                        oid: deelContractOid,
                    },
                    useMaster: false,
                });

                if (contract) {
                    data.contract = contract;
                }
            }
        }

        if (typeof data?.scopeOfWork === 'object' && data?.scopeOfWork !== null) {
            if (data.scopeOfWork.seniorityId && data.scopeOfWork.seniorityId !== '') {
                data.scopeOfWork.Seniority = await this._getSeniority(data.scopeOfWork.seniorityId);
            }

            if (data.scopeOfWork.scopeId && data.scopeOfWork.scopeId !== '') {
                data.scopeOfWork.Scope = await this._getScope(data.scopeOfWork.scopeId);
            }
        }

        data.isCancellationAllowed = isCancellationAllowed;

        return data;
    }

    async getPEOContractDetailsForEmployments({employments, getWorkLocationState = false, fallback = {}}) {
        const employmentsWithPeo = [];
        for (const employment of employments) {
            if (employment.hiring_type === HiringTypesEnum.peo) {

    async cancelEmploymentOnContractDeactivation(deelContractOid, desiredContractEndDate, finalPaymentDate = null) {
        const data: {
            payrollLegalEntityId: number;
            organizationId: number;
            employmentTerms: unknown;
            id: string;
        }[] = await EMSEmploymentsService.searchEmployments({
            contractOids: [deelContractOid],
            includeActiveTerm: true,
        });

        if (!data || !Array.isArray(data) || data.length === 0) {
            this.log.info({message: `No employment found for: ${deelContractOid}`});
            return;
        }

        const [employment] = data;

        await EMSEmploymentsService.updateEmploymentStatus({
            contractOids: [deelContractOid],
            // @ts-expect-error -- todo: fix types for updateEmploymentStatus
            status: EMSEmploymentsService.statuses.ENDED,
        });

        if (employment && employment.employmentTerms) {
            const payrollEventIds = await this.getEmploymentPayrollEventIds(employment);

            if (payrollEventIds.length > 0) {
                const instance = PmsEmsSelectionService.selectServiceInstance({legalEntityId: employment.payrollLegalEntityId, models: ['Groups', 'Cycles']});
                const payrollEventsResponse = await instance.listPayrollEvents({
                    ids: payrollEventIds,
                    // we only have to remove it from the regular cycles, the offcyles are manual so no remove needed
                    cycleStates: ['REGULAR'],
                    startDate: moment(desiredContractEndDate).subtract(30, 'days').toISOString(),
                    limit: 500,
                    organizationId: employment.organizationId,
                });

                this.log.info({
                    message: 'Payroll events to remove from',
                    desiredContractEndDate,
                    payrollEvents: payrollEventsResponse?.events ?? [],
                });

                const payrollEvents = [];
                const filterFinalPaymentDate = finalPaymentDate || desiredContractEndDate;
                if (payrollEventsResponse?.events) {
                    payrollEventsResponse.events.map((event) => {
                        const isCycleAfterFinalPayDate = this._filterNextCycleEvents(deelContractOid, filterFinalPaymentDate, event);
                        if (isCycleAfterFinalPayDate) payrollEvents.push(event);
                    });
                }

                if (payrollEvents.length > 0) {
                    const isPMSEnabled = PmsEmsSelectionService.isPMSEnabled({legalEntityId: employment.payrollLegalEntityId, models: ['Cycles']});
                    const identifier = isPMSEnabled ? deelContractOid : employment.id;

                    for (const payrollEvent of payrollEvents) {
                        const response = await instance.manuallyRemoveEmploymentFromPayrollEvent({}, identifier, payrollEvent.id);

                        this.log.info({
                            message: `Payroll Event removal response: ${response}`,
                            payrollEventId: payrollEvent.id,
                            employment: employment.id,
                        });
                    }
                }
            }
        }
    }

    private async getEmploymentPayrollEventIds(employment) {
        const isPMSEnabled = PmsEmsSelectionService.isPMSEnabled({legalEntityId: employment.payrollLegalEntityId, models: ['EmploymentCycles', 'Cycles']});
        let payrollEventIds = [];

        if (isPMSEnabled) {
            // For PMS: Use Employment Cycle Association Service to get all cycles for this employment
            const employmentCycleAssociations = await PmsEmploymentCycleAssociationService.getEmploymentCycleAssociations({
                filter: {
                    employmentId: [employment.contractOid],
                },
            });

            payrollEventIds = employmentCycleAssociations.map((association) => association.cycleId);
        } else {
            // For EMS: Use the existing method
            const employmentPayrollEventsMappingCount = await EMSEmploymentPayrollEventsService.countEmployeePayrollEvents({
                employmentIds: [employment.id],
            });

            payrollEventIds = Object.keys(employmentPayrollEventsMappingCount);
        }

        return payrollEventIds;
    }

    _filterNextCycleEvents(deelContractOid, desiredEndDate, payrollEvent) {
        if (!payrollEvent?.payrollCalendar?.payDate) {
            this.log.error({
                message: `the payrollEvent ${payrollEvent.id} does not have a pay date. It is not possible to check for exclusion in the termination flow for contractOid: ${deelContractOid}`,
            });
            return false;
        }

        const payrollPayDateIsBiggerThatTerminationFinalPayDate = moment(payrollEvent?.payrollCalendar?.payDate)
            .startOf('day')
            .isAfter(moment(desiredEndDate).startOf('day'));
        return payrollPayDateIsBiggerThatTerminationFinalPayDate;
    }

    /**
     * Updates a PEO contract with I-9 data from I9Section1
     * @param profileId - The profile ID to find the contract (used for fallback)
     * @param i9Data - The I-9 data to sync (legalStatus, arnUscisNumber, etc.)
     * @param contractOid - Optional: Specific contract OID to update. If not provided,
     *                      uses getDefaultContract which returns the first active contract.
     *                      IMPORTANT: During entity transfers, always pass contractOid to
     *                      ensure the correct (new) contract is updated.
     */
    async updateDefaultPEOContractWithI9({
        legalStatus,
        arnUscisNumber,
        i94Number,
        authorizedToWorkUntil,
        passportNumber,
        passportIssuingCountry,
        firstName,
        lastName,
        profileId,
        contractOid,
    }) {
        this.log.info({message: 'Updating PEO contract with i9 document info', data: {legalStatus, profileId, contractOid}});

        let contract;
        if (contractOid) {
            contract = await this.db.Contract.findFirst({
                where: {
                    oid: contractOid,
                    deletedAt: null,
                },
                attributes: ['id', 'clientLegalEntityId', 'HrisProfileId'],
                raw: true,
                useMaster: false,
            });
            if (!contract) {
                throw new Error(`Contract not found with oid: ${contractOid}`);
            }
            this.log.info({message: 'Using explicit contractOid for I-9 update', data: {contractOid}});
        } else {
            contract = await this.getDefaultContract(profileId, ['id', 'clientLegalEntityId', 'HrisProfileId']);
            this.log.info({message: 'Using getDefaultContract for I-9 update (no contractOid provided)', data: {profileId, foundContractId: contract?.id}});
        }

        await this.updateNamesIfChangedOnHrisProfile(contract.HrisProfileId, {
            firstName,
            lastName,
        });

        const payload = {
            deelContractId: String(contract.id),
            deelEntityId: String(contract.clientLegalEntityId),
            citizenStatus: this.mapCitizenStatus(legalStatus),
            ...(arnUscisNumber && {i9AlienRegistrationNumber: arnUscisNumber}),
            ...(authorizedToWorkUntil && {i9AlienRegistrationExpiration: moment(authorizedToWorkUntil).format('YYYY-MM-DD')}),
            ...(i94Number && {i94AdmissionNumber: i94Number}),
            ...(passportNumber && {i9ForeignPassportNumber: passportNumber}),
            ...(passportIssuingCountry && {i9PassportIssuingCountry: passportIssuingCountry}),
        };

        return await this.updatePEOContract(payload);
    }

    async validateDataAgainstPlatform(contract, dataToValidate) {
        const platformData = await this.getPlatformDataByContractType(contract);

        if (!platformData) return [];

        const mismatchedFields = [];

        for (const field in dataToValidate) {
            if (Object.hasOwnProperty.call(dataToValidate, field) && Object.hasOwnProperty.call(platformData, field)) {
                if (dataToValidate[field] !== platformData[field]) {
                    mismatchedFields.push({
                        field,
                        currentValue: platformData[field],
                    });
                }
            }
        }

        return mismatchedFields;
    }

    async updateEmployeeData({contract, employeeData, profile}) {
        const {firstName, middleName, lastName, preferredName, street, city, state, zip, birthDate, ssn, email, phone, maritalStatus, gender} = employeeData;

        // For each contract type it is necessary an implementation
        switch (contract.contractType) {
            case CONTRACT_TYPES.PEO:
                await this.db.sequelize.transaction(async (transaction) => {
                    contract = await this.getContractByOid({
                        deelContractOid: contract.oid,
                        addPEOData: true,
                        additionalAttributes: ['id', 'oid', 'ContractorId', 'HrisProfileId', 'jobTitleName'],
                        transaction,
                    });

                    const hrisProfile = await this.db.HrisProfile.findFirst({
                        where: {id: contract.HrisProfileId},
                        attributes: ['id'],
                        include: {model: this.db.HrisProfileAddresses, attributes: ['addressId'], required: false},
                        transaction,
                        useMaster: Boolean(transaction),
                    });

                    if (!hrisProfile) {
                        throw new HttpError.NotFound('HrisProfile not found');
                    }

                    const dataToUpdateContract = {
                        ...(firstName && lastName
                            ? {name: contract.jobTitleName ? `${firstName} ${lastName} - ${contract.jobTitleName}` : `${firstName} ${lastName}`}
                            : {}),
                        ...(state ? {state} : {}),
                    };

                    if (Object.entries(dataToUpdateContract).length > 0) {
                        await this.db.Contract.update(dataToUpdateContract, {
                            where: {
                                oid: contract.oid,
                            },
                            transaction,
                        });
                    }

                    const dataToUpdateProfile = {
                        ...(birthDate ? {dob: birthDate} : {}),
                        ...(email ? {email} : {}),
                        ...(phone ? {phone} : {}),
                        ...(ssn ? {company: {personal: {...(profile.company?.personal ?? {}), ssn}}} : {}),
                    };

                    if (Object.entries(dataToUpdateProfile).length > 0) {
                        await this.db.Profile.update(dataToUpdateProfile, {
                            where: {
                                id: profile.id,
                            },
                            transaction,
                        });
                    }

                    const dataToUpdateHrisProfile = {
                        ...(firstName ? {firstName} : {}),
                        ...(middleName ? {middleName} : {}),
                        ...(lastName ? {lastName} : {}),
                        ...(preferredName ? {preferredName} : {}),
                        ...(email ? {email} : {}),
                        ...(email ? {workEmail: email} : {}),
                        ...(birthDate ? {dateOfBirth: birthDate} : {}),
                    };

                    if (Object.entries(dataToUpdateHrisProfile).length > 0) {
                        await this.db.HrisProfile.update(dataToUpdateHrisProfile, {
                            where: {
                                id: contract.HrisProfileId,
                            },
                            transaction,
                        });
                    }

                    const dataToUpdateAddress: Record<string, unknown> = {
                        ...(street ? {street} : {}),
                        ...(city ? {city} : {}),
                        ...(state ? {state} : {}),
                        ...(zip ? {zip} : {}),
                    };

                    if (Object.entries(dataToUpdateAddress).length > 0) {
                        const matchedAddress = await peoLocationService.validateAndGetEntityGeoCode({
                            city,
                            geoCode: null,
                            zipCode: zip,
                        });

                        if (matchedAddress) {
                            dataToUpdateAddress.geoCode = matchedAddress.geoCode;
                        }

                        const hrisProfileAddress = await this.db.HrisProfileAddresses.findFirst({
                            where: {hrisProfileId: hrisProfile.id},
                            attributes: ['addressId'],
                            transaction,
                        });
                        if (hrisProfileAddress) {
                            await this.db.Address.update(dataToUpdateAddress, {where: {id: hrisProfileAddress.addressId}, transaction});
                        }
                    }

                    const dataToUpdatePeoContract = {
                        ...(firstName ? {firstName} : {}),
                        ...(lastName ? {lastName} : {}),
                        ...(street ? {addressLine1: street} : {}),
                        ...(city ? {city} : {}),
                        ...(state ? {stateCode: state} : {}),
                        ...(zip ? {zipCode: zip} : {}),
                        ...(ssn ? {ssn} : {}),
                        ...(birthDate ? {birthDate} : {}),
                        ...(maritalStatus ? {maritalStatus} : {}),
                        ...(gender ? {gender} : {}),
                    };

                    if (isHrisProfileEnrichmentEnabled() && gender) {
                        const demographicsService = new DemographicsService(this.db);
                        await demographicsService.upsertBasicDemographicsWithPeoDemographics(hrisProfile.id, {gender}, transaction);
                    }

                    if (Object.entries(dataToUpdatePeoContract).length > 0) {
                        await this.updatePEOContract({
                            deelContractId: contract.id,
                            ...dataToUpdatePeoContract,
                        });
                    }

                    const dataToUpdateEmploymentEmployee = {
                        ...(firstName ? {firstName} : {}),
                        ...(lastName ? {lastName} : {}),
                        ...(birthDate ? {birthDate} : {}),
                        ...(phone ? {mobilePhone: phone} : {}),
                    };

                    if (Object.entries(dataToUpdateEmploymentEmployee).length > 0) {
                        await employmentService.upsertEmployee({
                            profileId: profile.id,
                            ...dataToUpdateEmploymentEmployee,
                        });
                    }

                    if (isPEOEmploymentEventEnabled()) {
                        await peoEmploymentEventService.publishEmploymentUpdated({
                            id: contract.oid,
                            employeeName: [firstName, middleName, lastName]
                                .filter((n) => Boolean(n))
                                .join(' ')
                                .trim(),
                        });
                    }
                });
                return;
            default:
                return;
        }
    }

    async mapPeoContractList({contracts}) {
        const peoContracts = contracts.filter((contract) => contract.contractType === CONTRACT_TYPES.PEO);

        const additionalInfoMap = {};
        for (const contract of peoContracts) {
            try {
                additionalInfoMap[contract.oid] = await this._getPEOAndEmploymentInfo(contract.oid);
            } catch (err) {
                this.log.error({message: 'Failed to get additional info for PEO contract', contract});
                if (!utils.isLocal) {
                    throw err;
                }
            }
        }

        return contracts.map((contract) => {
            if (contract.contractType === CONTRACT_TYPES.PEO) {
                const peo = additionalInfoMap[contract.oid];

                return {
                    ...contract,
                    client: {id: contract.clientLegalEntityId, name: contract.clientLegalEntityName},
                    salary: peo?.payRate,
                    cycleScale: this.mapPublicPayPeriod(peo?.payPeriod),
                    type: contract.contractType,
                    peoEmployment: peo?.employment,
                };
            }
            return contract;
        });
    }

    async _getNextPaymentEventBasedOnPayslip(lastPayslip, contractOid) {
        const contract = await this.db.Contract.findFirst({
            where: {
                oid: contractOid,
            },
            attributes: ['id', 'clientLegalEntityId'],
            useMaster: false,
        });

        const instance = PmsEmsSelectionService.selectServiceInstance({legalEntityId: contract?.clientLegalEntityId, models: ['Cycles']});
        const commonEventWhereParams = {contractOid, limit: 1, type: 'REGULAR' as const, orderBy: 'start', order: 'asc' as const};
        if (!lastPayslip) {
            const response = await instance.findManyCycles({...commonEventWhereParams});
            return response?.data?.[0];
        }

        const payslipEventResponse = lastPayslip.payrollEventExternalId ? await instance.findManyCycles({ids: [lastPayslip.payrollEventExternalId]}) : null;
        const payslipEvent = payslipEventResponse?.data?.[0];
        const relativeEndDate = payslipEvent?.end ?? lastPayslip.to;

        //@ts-expect-error Incorrect value types in Payslip
        if (lastPayslip.status === this.db.Payslip.STATUSES.PUBLISHED && lastPayslip.to) {
            const nextEventResponse = await instance.findManyCycles({startFrom: relativeEndDate, ...commonEventWhereParams});
            return nextEventResponse?.data?.[0];
        }

        if (payslipEvent) {
            return payslipEvent;
        }

        const estimatedEventDuringPayslipResponse = await instance.findManyCycles({endFrom: lastPayslip.from, ...commonEventWhereParams});
        return estimatedEventDuringPayslipResponse?.data?.[0];
    }

    async getUpcomingPayrollCalendar(deelContractOid) {
        const contract = await this.db.Contract.findFirst({
            attributes: ['id', 'clientLegalEntityId'],
            where: {
                oid: deelContractOid,
            },
            useMaster: false,
            raw: true,
        });

        const employment = await this.getEmploymentInfo(contract.id);
        const employmentPayrollSettingId = employment?.payrollSettings?.id ?? null;
        if (!employmentPayrollSettingId) {
            this.log.error({message: "EE doesn't have associated payroll settings"});
            return null;
        }
        const currentTaxYear = new Date().getFullYear(),
            today = moment().startOf('day');

        const instance = PmsEmsSelectionService.selectServiceInstance({
            legalEntityId: contract.clientLegalEntityId,
            models: [{model: 'Calendars', type: ['REGULAR']}, 'Groups'],
        });
        let payrollCalendars = await instance.getByPayrollSettingsIdAndYear({
            payrollSettingsId: employmentPayrollSettingId,
            onlyForActivePayrollSettings: true,
            taxYear: currentTaxYear,
        });

        if (!payrollCalendars || payrollCalendars.length === 0) {
            const currentMonth = today.month() + 1;
            if (currentMonth !== 12) {
                this.log.error({message: "EE doesn't have payroll calendars for current year"});
                return null;
            }
            const nextTaxYear = currentTaxYear + 1;
            payrollCalendars = await instance.getByPayrollSettingsIdAndYear({
                payrollSettingsId: employmentPayrollSettingId,
                onlyForActivePayrollSettings: true,
                taxYear: nextTaxYear,
            });
            if (!payrollCalendars || payrollCalendars.length === 0) {
                this.log.error({message: "EE doesn't have payroll calendars for current and next year"});
                return null;
            }
        }

        const payrollCalendarsWithPayslip = payrollCalendars.filter((item) => {
            return item.payslipsPublishedDate;
        });

        payrollCalendarsWithPayslip.sort((a, b) => (moment(a.payDate).isBefore(b.payDate) ? -1 : 1));

        return payrollCalendarsWithPayslip.find((calendar) => moment(calendar.payDate).isSameOrAfter(today));
    }

    async getEmploymentTerms(deelContractOid, payrollSettingsId?: string) {

            contractOids: [deelContractOid],
            includeAllTerms: true,
            includePayrollSettings: true,
            payrollSettingsId,
        });

            id: employments[0].id,
            employmentTerms: [],
            employmentPayrollSettings: employments[0].employmentPayrollSettings,
        };

        for (const term of employments[0].employmentTerms) {
            if (typeof term !== 'object' || term === null) {
                continue;
            }

            currentEmployment.employmentTerms.push({
                id: term.id,
                effectiveDate: term.effectiveDate,
                type: term.type,
                changeReason: term.changeReason,
                createdAt: term.createdAt,
                jobTitle: term.jobTitle,
                division: term.division,
                department: term.department,
                location: term.location,
                status: term.status,
                rate: term.rate,
            });
        }

        return currentEmployment;
    }

    async getEntitlement(deelContractOid) {
        // eslint-disable-next-line @typescript-eslint/no-var-requires -- avoid circular dependency
        const timeOffService = require('../time_off');

        const contract = await this.db.Contract.findFirst({
            attributes: ['id'],

            include: {

        });

        if (!contract?.HrisProfile?.oid) {
            this.log.warn({message: `Could not find Contract or HrisProfile for ${deelContractOid}`});
            return null;
        }

        const entitlementResponse = await timeOffService.getEntitlementsForProfiles([contract.HrisProfile.oid], {
            policyTypeName: PolicyTypeName.VACATION,
        });
        const entitlement = entitlementResponse?.entitlementsPerProfile[0]?.entitlements;

        if (!Array.isArray(entitlement) || entitlement.length === 0) {
            this.log.warn({message: `No entitlement found: ${entitlementResponse}`});
            return null;
        }

        const mostRecentEntitlement = entitlement.reduce(
            (a: {trackingPeriod: string}, b: {trackingPeriod: string}) => (moment(a?.trackingPeriod).isAfter(moment(b?.trackingPeriod)) ? a : b),
            entitlement[0]
        );

        return mostRecentEntitlement;
    }

    async getContractUnderwritingDetails({contractOid, type}) {
        const contract = await this.getContractByOid({deelContractOid: contractOid, additionalAttributes: ['clientLegalEntityId'], addPEOData: true});

        const legalEntity = await this.db.LegalEntity.findFirst({
            attributes: ['publicId'],
            where: {
                id: contract.clientLegalEntityId,
            },
            useMaster: false,
        });

        const {peoContract} = contract;

        const prismResourceRequests = await peoPrismResourceRequestService.getPrismResourceRequestByLegalEntityId({
            legalEntityId: legalEntity.publicId,
            prismRequestStatus: PrismResourceRequestStatus.UNDER_REVIEW,
        });

        const positionRequest = prismResourceRequests.find(
            (request) => request.resource === PrismResourceRequestResource.POSITION && request.description === peoContract.jobCode
        );

        const workLocationRequest = prismResourceRequests.find(
            (request) => request.resource === PrismResourceRequestResource.WORK_LOCATION && request.description === peoContract.workLocationCode
        );

        const contractHasSomeRequestInUnderReview = Boolean(positionRequest || workLocationRequest);

        if (!contractHasSomeRequestInUnderReview) {
            return null;
        }

        const contractMissingUpdates = await peoContractMissingUpdateService.getPeoContractMissingUpdateByPeoContractIdByFilter({
            deelContractId: contract.id,
            filter: {
                type,
            },
        });

        const getWorkLocationChanges = async ({legalEntityId, oldValue, updatedValue}) => {
            const previousPeoWorkLocation = oldValue && (await peoWorkLocationService.getPEOWorkLocationByPrismWorkLocationId(legalEntityId, oldValue));
            const updatedPeoWorkLocation = updatedValue && (await peoWorkLocationService.getPEOWorkLocationByPrismWorkLocationId(legalEntityId, updatedValue));

            if (!previousPeoWorkLocation && !updatedPeoWorkLocation) {
                return;
            }

            oldValue =
                previousPeoWorkLocation &&
                (await this.db.EntityWorkLocation.findFirst({
                    useMaster: false,
                    attributes: [['public_id', 'id'], 'name', 'isWfh'],
                    where: {
                        publicId: previousPeoWorkLocation.deelEntityWorkLocationId,
                    },
                    include: [{model: this.db.Address, as: 'PhysicalAddress', attributes: ['country', 'state', 'city', 'street', 'zip']}],
                }));

            updatedValue =
                updatedPeoWorkLocation &&
                (await this.db.EntityWorkLocation.findFirst({
                    useMaster: false,
                    attributes: [['public_id', 'id'], 'name', 'isWfh'],
                    where: {
                        publicId: updatedPeoWorkLocation.deelEntityWorkLocationId,
                    },
                    include: [{model: this.db.Address, as: 'PhysicalAddress', attributes: ['country', 'state', 'city', 'street', 'zip']}],
                }));

            return {oldValue, updatedValue};
        };

        const getJobTitleChanges = async ({legalEntityId, oldValue, updatedValue}) => {
            let clientCodes = [];
            try {
                clientCodes = await peoClientMasterService.getPEOJobCodesByEntityId(legalEntityId);
            } catch (error) {
                this.log.info({message: 'Error getting job codes', legalEntityId, error});
            }

            const oldPosition = clientCodes?.find((jc) => jc.id === oldValue);
            const updatedPosition = clientCodes?.find((jc) => jc.id === updatedValue);

            oldValue &&= {code: oldPosition?.id, title: oldPosition?.jobTitle ?? oldValue};
            updatedValue = {code: updatedPosition?.id, title: updatedPosition?.jobTitle ?? updatedValue};

            return {oldValue, updatedValue};
        };

        const results = [];

        if (positionRequest) {
            const missingUpdate = contractMissingUpdates.find(
                (update) => update.type === PeoContractMissingUpdateType.JOB_TITLE && update.updatedValue === peoContract.jobCode
            );

            const changes = await getJobTitleChanges({
                legalEntityId: contract.clientLegalEntityId,
                oldValue: missingUpdate?.oldValue,
                updatedValue: peoContract.jobCode,
            });

            const updates = {
                ...(missingUpdate ?? {}),
                ...changes,
                type: PeoContractMissingUpdateType.JOB_TITLE,
                createdAt: peoContract.createdAt,
                updatedAt: peoContract.updatedAt,
                peoContractId: peoContract.id,
                updateRequestedAt: peoContract.peoStartDate,
                status: positionRequest.status,
            };

            results.push(updates);
        }

        if (workLocationRequest) {
            const missingUpdate = contractMissingUpdates.find(
                (update) => update.type === PeoContractMissingUpdateType.WORK_LOCATION && update.updatedValue === peoContract.workLocationCode
            );

            const changes = await getWorkLocationChanges({
                legalEntityId: contract.clientLegalEntityId,
                oldValue: missingUpdate?.oldValue,
                updatedValue: peoContract.workLocationCode,
            });

            const updates = {
                ...(missingUpdate ?? {}),
                ...normalizeKeys(changes ?? {}),
                type: PeoContractMissingUpdateType.WORK_LOCATION,
                createdAt: peoContract.createdAt,
                updatedAt: peoContract.updatedAt,
                peoContractId: peoContract.id,
                updateRequestedAt: peoContract.peoStartDate,
                status: workLocationRequest.status,
            };

            results.push(updates);
        }

        return results;
    }

    async moveComplianceDocuments({fromContractOid, toContractOid}) {
        const fromContract = await this.getContractByOid({
            deelContractOid: fromContractOid,
            additionalAttributes: ['clientLegalEntityId', 'contractorId', 'status'],
        });

        if (!fromContract) {
            throw new HttpError.NotFound(`Contract ${fromContractOid} not found`);
        }

        if (!TERMINATED_CONTRACT_STATUSES.includes(fromContract.status)) {
            throw new HttpError.BadRequest(`Contract ${fromContractOid} should be terminated`);
        }

        const toContract = await this.getContractByOid({
            deelContractOid: toContractOid,
            additionalAttributes: ['clientLegalEntityId', 'contractorId', 'status'],
        });

        if (!toContract) {
            throw new HttpError.NotFound(`Contract ${toContractOid} not found`);
        }

        if (fromContract.id === toContract.id) {
            throw new HttpError.BadRequest('Contracts should not be the same');
        }

        if (fromContract.clientLegalEntityId !== toContract.clientLegalEntityId) {
            throw new HttpError.BadRequest('Contracts should be from the same legal entity');
        }

        if (fromContract.contractorId !== toContract.contractorId) {
            throw new HttpError.BadRequest('Contracts should be from the same contractor');
        }

        const {data, status} = await this.client.post(PEOEndpoints.moveComplianceDocuments, {
            fromContractId: fromContract.id,
            toContractId: toContract.id,
            deelOwnerProfileId: fromContract.contractorId,
        });

        if (status !== 200 && status !== 204) {
            throw new HttpError(status, `Failed to move compliance documents from ${fromContractOid} to ${toContractOid} with ${data.message}`);
        }

        return data;
    }

    async getContractRequirementIdsByAgreementTypes(agreementTypes: string[]): Promise<number[]> {
        const agreementTypeToTitleMap: Record<string, string> = {
            /* eslint-disable @typescript-eslint/naming-convention -- Keys match AgreementType enum values */
            ARBITRATION_AGREEMENT: 'Arbitration agreement',
            WSE_NOTICE_OF_PEO_RELATIONSHIP: 'WSE notice of PEO relationship',
            /* eslint-enable @typescript-eslint/naming-convention -- Re-enable after AgreementType keys */
        };

        if (!agreementTypes || agreementTypes.length === 0) {
            return [];
        }

        const titles = agreementTypes.map((type) => agreementTypeToTitleMap[type]).filter((title): title is string => title !== undefined);

        if (titles.length === 0) {
            return [];
        }

        const requirements = await this.db.ContractRequirement.findAll({
            attributes: ['id'],
            where: {
                title: {[this.db.Sequelize.Op.in]: titles},
                country: 'US',
            },
            raw: true,
            useMaster: true,
        });

        return requirements.map((r) => r.id);
    }

                where: {
                    contractId: baseContract.id,
                },
                useMaster: false,
            });

            if (i9Section1BaseContract) {
                const existingI9Section1ForRehireContract = await this.db.I9Section1.findFirst({
                    where: {
                        contractId: rehireContract.id,
                    },
                    useMaster: false,
                });

                if (!existingI9Section1ForRehireContract) {
                    await this.db.I9Section1.create({
                        ...i9Section1BaseContract.dataValues,
                        contractId: rehireContract.id,
                    });

                    await this.syncI9InfoToPEoContract({contractOid: rehireContract.oid, transaction: undefined});
                }
            }

            if (I9Section2BaseContract) {
                const existingI9Section2ForRehireContract = await this.db.I9Section2.findFirst({
                    where: {
                        contractId: rehireContract.id,
                    },
                    useMaster: false,
                });

                if (!existingI9Section2ForRehireContract) {
                    await this.db.I9Section2.create({
                        ...I9Section2BaseContract.dataValues,
                        contractId: rehireContract.id,
                    });
                }
            }

            if (I9RepresentativeInvitationBaseContract) {
                const existingI9RepresentativeInvitationForRehireContract = await this.db.I9RepresentativeInvitation.findFirst({
                    where: {
                        contractId: rehireContract.id,
                    },
                    useMaster: false,
                });

                if (!existingI9RepresentativeInvitationForRehireContract) {
                    await this.db.I9RepresentativeInvitation.create({
                        ...I9RepresentativeInvitationBaseContract.dataValues,
                        contractId: rehireContract.id,
                    });
                }
            }
        }
    }

    async saveWorkRecordsSidecar({contractId, payload, transaction, employmentType, workHoursPerWeek, seniorityId, withContractUpdates = false}) {
        const sidecarService = new PeoSidecarService({db: this.db});
        const workType = sidecarService.getWorkType({employmentType});
        const ftePercentage = sidecarService.getFtePercentage({workType, weeklyHours: workHoursPerWeek});
        let seniorityLevel;
        if (seniorityId) {
            seniorityLevel = (await this._getSeniority(seniorityId))?.name;
        }

        payload = {
            workStatus: WorkStatusEnum.ACTIVE,
            ...payload,
            ...(workType ? {workType} : {}),
            ...(ftePercentage ? {ftePercentage} : {}),
            ...(seniorityLevel ? {seniorityLevel} : {}),
        };

        await sidecarService.saveSidecar({contractId, payload, transaction, withContractUpdates});
    }
}

let peoContractServiceInstance: PEOContractService | null = null;

function getPeoContractServiceInstance(): PEOContractService {
    if (!peoContractServiceInstance) {
        peoContractServiceInstance = new PEOContractService(null);
    }
    return peoContractServiceInstance;
}

// Define lazy getter for the service instance
// This avoids circular dependency by delaying instantiation until first access
Object.defineProperty(exports, 'peoContractService', {
    get() {
        return getPeoContractServiceInstance();
    },
    enumerable: true,
    configurable: true,
});

// TypeScript type-only export declaration for IDE support
export declare const peoContractService: PEOContractService;
