function generateRequestId() {
  return `fetch-${Date.now()}-${Math.random().toString(36).substr(2, 9)}`;
}

function makeRequest(cursor, requestId) {
  return new Promise((resolve, reject) => {
    let path = `/peo_integration/hris_integration/${CONFIG.integrationId}/employees`;
    if (cursor) {
      path += `?cursor=${encodeURIComponent(cursor)}`;
    }

    const options = {
      hostname: 'api-admin.demo.deel.com',
      path: path,
      method: 'GET',
      headers: {
        'accept': 'application/json',
        'x-api-version': '2',
        'x-auth-token': CONFIG.authToken,
        'x-request-id': requestId
      }
    };

    const req = https.request(options, (res) => {
      let data = '';
      res.on('data', (chunk) => data += chunk);
      res.on('end', () => {
        try {
          const json = JSON.parse(data);
          resolve({ data: json, statusCode: res.statusCode, headers: res.headers });
        } catch (e) {
          reject(new Error(`Failed to parse JSON: ${e.message}`));
        }
      });
    });

    req.on('error', reject);
    req.end();
  });
}

async function fetchAllEmployees() {
  const allEmployees = [];
  const requestIds = [];
  const responses = [];
  let cursor = null;
  let page = 1;

  console.log('Starting to fetch employees...\n');

  while (true) {
    const requestId = generateRequestId();
    requestIds.push(requestId);

    console.log(`Page ${page}: Request ID = ${requestId}`);

    try {
      const result = await makeRequest(cursor, requestId);
      const { data, statusCode } = result;

      if (statusCode !== 200) {
        console.error(`Error: HTTP ${statusCode}`);
        console.error(JSON.stringify(data, null, 2));
        break;
      }

      const employees = data.employees || [];
      console.log(`  -> Fetched ${employees.length} employees`);

      responses.push({
        page,
        requestId,
        cursor: cursor || 'initial',
        employeesCount: employees.length,
        data: data
      });

      allEmployees.push(...employees);

      cursor = data.cursor;
      if (!cursor || employees.length === 0) {
        console.log('\nNo more pages to fetch.');
        break;
      }

      page++;
    } catch (error) {
      console.error(`Error on page ${page}:`, error.message);
      break;
    }
  }

  console.log(`\nTotal employees fetched: ${allEmployees.length}`);
  console.log(`Total requests made: ${requestIds.length}`);

  // Save results
  const output = {
    fetchedAt: new Date().toISOString(),
    totalEmployees: allEmployees.length,
    totalRequests: requestIds.length,
    integrationId: CONFIG.integrationId,
    requestIds: requestIds,
    employees: allEmployees,
    rawResponses: responses
  };

  fs.writeFileSync(CONFIG.outputFile, JSON.stringify(output, null, 2));
  console.log(`\nData saved to: ${CONFIG.outputFile}`);

  // Save request IDs separately for easy access
  fs.writeFileSync(CONFIG.requestIdsFile, JSON.stringify({
    fetchedAt: new Date().toISOString(),
    requestIds: requestIds
  }, null, 2));
  console.log(`Request IDs saved to: ${CONFIG.requestIdsFile}`);

  return { allEmployees, requestIds };
}

fetchAllEmployees().catch(console.error);
