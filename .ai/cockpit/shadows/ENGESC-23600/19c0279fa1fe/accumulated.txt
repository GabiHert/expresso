    @Joi((joi) => ({

        }),
    }))
    async getTransferResources({params, res}) {
        const {legalEntityPublicId} = params;

        try {
            const resources = await transferResourcesService.getTransferResources(legalEntityPublicId);
            return res.status(HTTP_CODES.ok).json(resources);
        } catch (error) {
            const statusCode =
                typeof error.statusCode === 'number' && error.statusCode >= 100 && error.statusCode <= 599 ? error.statusCode : HTTP_CODES.internalServerError;
            return res.status(statusCode).json({
                success: false,
                error: error.message,
                details: error.data || error.details,
            });
        }
    }

    @Post('/entity_transfer')
    @Middleware((app) => app.permissions('admin:contracts.write'))
    @Joi((joi) => ({
        body: joi
            .object()
            .keys({
                // ====== Resume mode fields ======
                transferItemId: joi.string().uuid(),
                // resumeFromStep is only valid in resume mode (when transferItemId is present)
                resumeFromStep: joi.string().min(1).max(100).when('transferItemId', {
                    is: joi.exist(),
                    then: joi.optional(),
                    otherwise: joi.forbidden(),
                }),

                // ====== Full payload mode fields ======
                // Transfer fields - Number IDs
                organizationId: joi.number().integer().min(1),
                // basePeoContractOid is peo_contracts.deel_contract_oid (employee number on PrismHR)
                basePeoContractOid: joi.string(),
                newTeamId: joi.number().integer().min(1).optional(),
                // Transfer fields - UUIDs
                newPayrollSettingsId: joi.string(), // employment.payroll_settings.id (TEXT - can be UUID or CUID)
                requesterProfilePublicId: joi.string().uuid(), // profile.public_id
                sourceLegalEntityPublicId: joi.string().uuid(),
                destinationLegalEntityPublicId: joi.string().uuid(),
                // Transfer fields - Date
                effectiveDate: joi.plainDateString(),
                // Transfer fields - String IDs with length limits
                agreementId: joi.string().min(1).max(100).optional(),
                // Transfer item fields - Prism IDs with database-aligned constraints
                newBenefitGroupId: joi.string().min(1).max(10),
                newPtoPolicyId: joi.string().uuid(), // time_off.policies.uid
                newWorkLocationId: joi.string().uuid(), // entity_work_locations.public_id
                newPositionPublicId: joi.string().uuid(), // peo_positions.public_id
                // New contract OID for resume scenarios (when contract was already created)
                newContractOid: joi.string().min(5).max(20).optional(),
            })
            .xor('transferItemId', 'organizationId') // Either resume mode OR full payload mode
            .with('organizationId', [
                'basePeoContractOid',
                'requesterProfilePublicId',
                'sourceLegalEntityPublicId',
                'destinationLegalEntityPublicId',
                'effectiveDate',
                'newBenefitGroupId',
                'newPayrollSettingsId',
                'newPtoPolicyId',
                'newWorkLocationId',
                'newPositionPublicId',
            ]), // Full payload requires all these fields
    }))
    async executeEntityTransfer({body, res}) {
        let transferId = 'unknown';
        let itemId = 'unknown';

        try {
            const isResumeMode = Boolean(body.transferItemId);
            const repository = new EntityTransferRepository();

            let transfer: PeoEmployeeTransfer;
            let transferItem: PeoEmployeeTransferItem;

            if (isResumeMode) {
                ({transfer, transferItem, transferId, itemId} = await this.handleResumeMode(body, repository));
            } else {
                const result = await this.handleFullPayloadMode(body, repository);
                ({transfer, transferItem, transferId, itemId} = result);
            }

            const entityTransferService = new EntityTransferService(repository);
            const result = await entityTransferService.executeTransfer(transfer, transferItem);

            if (result.success) {
                await repository.updateTransferItemStatus(itemId, TransferItemStatus.COMPLETED);
            } else {
                await repository.updateTransferItemStatus(itemId, TransferItemStatus.FAILED);
            }

            let errorDetails: unknown = result.error?.message || result.error?.toString();
            if (errorDetails) {
                try {
                    errorDetails = JSON.parse(errorDetails as string);
                } catch {
                    // If not JSON, keep as string
                }
            }

            const response = {
                success: result.success,
                transferId,
                itemId,
                status: result.status,
                completedSteps: result.completedSteps,
                crossHireCompleted: result.crossHireCompleted,
                workLocationId: transferItem.newWorkLocationId,
                positionPublicId: transferItem.newPositionPublicId,
                ...(result.newContractOid && {newContractOid: result.newContractOid}),
                ...(result.resumeFromStep && {resumeFromStep: result.resumeFromStep}),
                ...(errorDetails && {error: errorDetails}),
                ...(result.error && {
                    errorDetails: {
                        message: result.error.message,
                        stack: result.error.stack,
                        name: result.error.name,
                    },
                }),
                ...(result.rollbackErrors && {rollbackErrors: result.rollbackErrors}),
            };
            const hasErrors = (result.rollbackErrors && result.rollbackErrors.length > 0) || errorDetails;

            return res.status(hasErrors ? HTTP_CODES.unprocessable_entity : HTTP_CODES.ok).json(response);
        } catch (error) {
            if (error instanceof HttpError) {
                throw error;
            }

            // Parse Zod validation errors if present
            let errorDetails = error.message;
            try {
                errorDetails = JSON.parse(error.message);
            } catch {
                // If not JSON, keep as string
            }

            return res.status(HTTP_CODES.unprocessable_entity).json({
                success: false,
                transferId,
                itemId,
                error: errorDetails,
                errorMessage: error.message,
                errorName: error.name,
                stack: error.stack,
            });
        }
    }

    private async handleResumeMode(
        body: {transferItemId: string; resumeFromStep?: string},
        repository: EntityTransferRepository
    ): Promise<{transfer: PeoEmployeeTransfer; transferItem: PeoEmployeeTransferItem; transferId: string; itemId: string}> {
        log.info('[TechOps] Entity transfer resume mode', {
            transferItemId: body.transferItemId,
            resumeFromStep: body.resumeFromStep,
        });

        const result = await repository.loadTransferItemWithTransfer(body.transferItemId);

        if (!result) {
            throw new HttpError.NotFound(`Transfer item not found: ${body.transferItemId}`);
        }

        const transfer = result.transfer;
        const transferItem = result.item;
        const transferId = transfer.id;
        const itemId = transferItem.id;

        if (body.resumeFromStep) {
            transferItem.resumeFromStep = body.resumeFromStep;
        }

        log.info('[TechOps] Transfer item loaded for resume', {
            transferId,
            itemId,
            currentStatus: transferItem.status,
            resumeFromStep: transferItem.resumeFromStep,
        });

        return {transfer, transferItem, transferId, itemId};
    }

    private async handleFullPayloadMode(
        body: {
            organizationId: number;
            requesterProfilePublicId: string;
            sourceLegalEntityPublicId: string;
            destinationLegalEntityPublicId: string;
            effectiveDate: string;
            agreementId?: string;
            basePeoContractOid: string;
            newBenefitGroupId: string;
            newPayrollSettingsId: string;
            newPtoPolicyId: string;
            newTeamId?: number;
            newWorkLocationId: string;
            newPositionPublicId: string;
            newContractOid?: string;
        },
        repository: EntityTransferRepository
    ): Promise<{transfer: PeoEmployeeTransfer; transferItem: PeoEmployeeTransferItem; transferId: string; itemId: string}> {
        const {
            organizationId,
            requesterProfilePublicId,
            sourceLegalEntityPublicId,
            destinationLegalEntityPublicId,
            effectiveDate,
            agreementId,
            basePeoContractOid,
            newBenefitGroupId,
            newPayrollSettingsId,
            newPtoPolicyId,
            newTeamId,
            newWorkLocationId,
            newPositionPublicId,
            newContractOid,
        } = body;

        log.info('[TechOps] Entity transfer full payload mode', {
            organizationId,
            basePeoContractOid,
        });

        const transfer = await repository.createTransfer({
            organizationId,
            requesterProfilePublicId,
            sourceLegalEntityPublicId,
            destinationLegalEntityPublicId,
            effectiveDate,
            agreementId,
            items: [
                {
                    baseContractOid: basePeoContractOid,
                    newBenefitPrismGroupId: newBenefitGroupId,
                    newEmploymentPayrollSettingId: newPayrollSettingsId,
                    newPtoPolicyId,
                    newWorkLocationId,
                    newPositionPublicId,
                    newTeamId,
                },
            ],
        });

        const transferId = transfer.id;
        const transferItem = transfer.items![0];
        const itemId = transferItem.id;

        if (newContractOid) {
            transferItem.newContractOid = newContractOid;
        }

        log.info('[TechOps] Transfer created in PEO service', {
            transferId,
            itemId,
        });

        return {transfer, transferItem, transferId, itemId};
    }

    /**
     * Creates a missing employment for a PEO contract.
     * Use when a contract exists but employment was not created (e.g., entity transfer bug).
     *
     * The endpoint will:
     * 1. Check if employment already exists (skip if it does)
     * 2. Fetch all required data from the NEW contract and its peo_contracts record
     * 3. Use peo_contracts.peoStartDate for employment start date
     * 4. Resolve jobTitle from jobCode (fallback to jobCode if not found)
     * 5. Create employment via Employment microservice
     *
     * Example tickets: ENGESC-23650, ENGESC-23600
     */
    @Post('/contract/:oid/create_missing_employment')
    @Joi((joi, db) => ({
        params: joi.object().keys({
            oid: joi.publicId(db.Contract).permission('admin:contracts.write', db.Contract).secure().resolve(false).required(),
        }),
    }))
    async createMissingEmployment({params, res}) {
        const {oid: contractOid} = params;

        const result = await peoContractService.createMissingEmploymentForContract({
            contractOid,
        });

        return res.status(HTTP_CODES.ok).json(result);
    }
