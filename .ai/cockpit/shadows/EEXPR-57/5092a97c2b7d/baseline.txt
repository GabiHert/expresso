    describe('enrichPEOContractWithHrisProfileInfo', () => {
        let dbMock;
        let peoContract;
        let hrisProfileInfo;

        beforeEach(() => {
            dbMock = {
                HrisProfile: {
                    findFirst: jest.fn(),
                },
                Contract: {},
            };

            peoContract = {
                deelContractOid: faker.datatype.uuid(),

                firstName: 'Updated First Name',
                lastName: 'Updated Last Name',
                startDate: '2024-01-01',
            };

            jest.spyOn(peoContractService, 'db', 'get').mockReturnValue(dbMock);
        });

        it('should enrich PEO contract with HRIS profile info when profile exists', async () => {
            dbMock.HrisProfile.findFirst.mockResolvedValue(hrisProfileInfo);

            const result = await peoContractService.enrichPEOContractWithHrisProfileInfo(peoContract);

            expect(dbMock.HrisProfile.findFirst).toHaveBeenCalledWith({
                attributes: ['firstName', 'lastName', 'startDate'],
                include: [
                    {
                        model: dbMock.Contract,
                        attributes: [],
                        where: {
                            oid: peoContract.deelContractOid,
                        },
                        required: true,
                    },
                ],
                useMaster: false,
                raw: true,
            });

            expect(result.firstName).toBe(hrisProfileInfo.firstName);
            expect(result.lastName).toBe(hrisProfileInfo.lastName);
            expect(result.origHireDate).toBe(hrisProfileInfo.startDate);
            expect(result.lastHireDate).toBe(hrisProfileInfo.startDate);
        });

        it('should enrich PEO contract with custom attributes when provided', async () => {
            const customAttributes = ['firstName', 'lastName', 'email'];
            const peoContract = {
                deelContractOid: faker.datatype.uuid(),
                firstName: 'Original First Name',
                lastName: 'Original Last Name',
                origHireDate: '2023-01-01',
                lastHireDate: '2023-01-01',
            };

            const hrisProfileInfo = {
                firstName: 'Updated First Name',
                lastName: 'Updated Last Name',
            };

            dbMock.HrisProfile.findFirst.mockResolvedValue(hrisProfileInfo);

            const result = await peoContractService.enrichPEOContractWithHrisProfileInfo(peoContract, customAttributes);

            expect(dbMock.HrisProfile.findFirst).toHaveBeenCalledWith({
                attributes: customAttributes,
                include: [
                    {
                        model: dbMock.Contract,
                        attributes: [],
                        where: {
                            oid: peoContract.deelContractOid,
                        },
                        required: true,
                    },
                ],
                useMaster: false,
                raw: true,
            });
            expect(result.firstName).toBe(hrisProfileInfo.firstName);
            expect(result.lastName).toBe(hrisProfileInfo.lastName);
            expect(result.origHireDate).toBe(peoContract.origHireDate);
        });

        it('should keep original values when HRIS profile info is null', async () => {
            dbMock.HrisProfile.findFirst.mockResolvedValue(null);

            const result = await peoContractService.enrichPEOContractWithHrisProfileInfo(peoContract);

            expect(result.firstName).toBe(peoContract.firstName);
            expect(result.lastName).toBe(peoContract.lastName);
            expect(result.origHireDate).toBe(peoContract.origHireDate);
            expect(result.lastHireDate).toBe(peoContract.lastHireDate);

        it('should keep original values when HRIS profile fields are null/undefined', async () => {
            const hrisProfileWithNullFields = {

                lastName: undefined,

            expect(result.lastHireDate).toBe(peoContract.lastHireDate);

        it('should return original contract when database query throws error', async () => {
            const error = new Error('Database connection failed');

            const result = await peoContractService.enrichPEOContractWithHrisProfileInfo(peoContract);

            expect(result).toEqual(peoContract);
            expect(result.firstName).toBe(peoContract.firstName);
            expect(result.lastName).toBe(peoContract.lastName);
            expect(result.origHireDate).toBe(peoContract.origHireDate);
            expect(result.lastHireDate).toBe(peoContract.lastHireDate);

                    contractOid: mockPeoContracts[0].deelContractOid,
                    data: {
                        rate: 105000,
                        scale: 'YEAR',
                    },
                    effectiveDate: '2025-01-01',
                },
            ];

            gpContractDataChangesService.Salary.getActiveDataForContract.mockResolvedValue(mockActiveSalaryChanges);

            const result = await peoContractService.getPEOContractsByFilter({deelEntityIds: ['123']}, {includeActiveSalary: true});

            // Only first contract should have payroll settings
            expect(result[0].payrollSettings).not.toBeNull();
            expect(result[1].payrollSettings).toBeNull();

            // Only first contract OID should be included in the query
            expect(gpContractDataChangesService.Salary.getActiveDataForContract).toHaveBeenCalledWith({
                contractOids: [mockPeoContracts[0].deelContractOid],
            });

            // First contract should have compensation details
            expect(result[0].currentCompensationDetails).toEqual({
                rate: 105000,
                scale: 'YEAR',
                effectiveDate: '2025-01-01',
                workHoursPerWeek: mockPeoContracts[0].workHoursPerWeek,
            });

            // Second contract should have null compensation details
            expect(result[1].currentCompensationDetails).toEqual({
                rate: null,
                scale: null,
                effectiveDate: null,
                workHoursPerWeek: null,
            });
        });

        it('should handle errors gracefully when processing salary data', async () => {
            const mockPayrollSettingsIds = [{id: 'ps-1'}];
            const mockPayrollSettings = {id: 'ps-1', name: 'Test Payroll Settings'};

            jest.spyOn(EMSPayrollSettingsService, 'getPayrollSettingsForContractOid').mockResolvedValue(mockPayrollSettingsIds);
            jest.spyOn(EMSPayrollSettingsService, 'getById').mockResolvedValue(mockPayrollSettings);

            // Create invalid salary data that will throw an error
            const mockActiveSalaryChanges = [
                {
                    contractOid: mockPeoContracts[0].deelContractOid,
                    // Missing 'data' property to cause an error
                },
            ];

            gpContractDataChangesService.Salary.getActiveDataForContract.mockResolvedValue(mockActiveSalaryChanges);

            const result = await peoContractService.getPEOContractsByFilter({deelEntityIds: ['123']}, {includeActiveSalary: true});

            // Should still set compensation details with null values
            expect(result[0].currentCompensationDetails).toEqual({
                rate: null,
                scale: null,
                effectiveDate: null,
                workHoursPerWeek: 40,
            });
        });

        it('should cache payroll settings when multiple contracts have the same payroll setting', async () => {
            const mockPayrollSettingsIds = [{id: 'ps-1'}];
            const mockPayrollSettings = {
                id: 'ps-1',
                name: 'Test Payroll Settings',
            };

            jest.spyOn(EMSPayrollSettingsService, 'getPayrollSettingsForContractOid').mockResolvedValue(mockPayrollSettingsIds);
            const getByIdSpy = jest.spyOn(EMSPayrollSettingsService, 'getById').mockResolvedValue(mockPayrollSettings);

            await peoContractService.getPEOContractsByFilter({deelEntityIds: ['123']}, {includePayrollSettings: true});

            // Should only call getById once even though we have 2 contracts with the same payroll settings
            expect(getByIdSpy).toHaveBeenCalledTimes(1);
            expect(getByIdSpy).toHaveBeenCalledWith('ps-1');
        });

        it('should not call salary service when no contracts have payroll settings and includeActiveSalary is true', async () => {
            jest.spyOn(EMSPayrollSettingsService, 'getPayrollSettingsForContractOid').mockResolvedValue([]);

            await peoContractService.getPEOContractsByFilter({deelEntityIds: ['123']}, {includeActiveSalary: true});

            // Should not call the salary service since no contracts have payroll settings
            expect(gpContractDataChangesService.Salary.getActiveDataForContract).not.toHaveBeenCalled();
        });
    });
