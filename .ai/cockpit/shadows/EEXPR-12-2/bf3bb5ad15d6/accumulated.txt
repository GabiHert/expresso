import { PeoEmployeeTransfer, PeoEmployeeTransferItem, PeoEmployeeTransferSignature, TransferItemStatus, TransferStatus } from '@models/entityTransfer';
import { PeoContract } from '@models/peoContract/peoContract';
import { Op, Transaction, WhereOptions } from 'sequelize';
import { Service } from 'typedi';

export interface CreateTransferInput {
    organizationId: number;
    requesterProfilePublicId: string; // UUID - profile.public_id
    sourceLegalEntityPublicId: string;
    destinationLegalEntityPublicId: string;
    effectiveDate: Date;
    agreementId?: string;
    items: CreateTransferItemInput[];
}

export interface CreateTransferItemInput {
    baseContractOid: string;
    newBenefitPrismGroupId: string;
    newEmploymentPayrollSettingId: string;
    newPtoPolicyId: string;
    newWorkLocationId: string;
    newPositionPublicId: string;
    newTeamId?: number;
}

export interface UpdateTransferItemInput {
    status?: TransferItemStatus;
    newContractOid?: string;
    resumeFromStep?: string;
}

export interface TransferItemResponse {
    id: string;
    status: string;
    baseContractOid: string;
    deelContractId: number | null;
    employeeName: string;
    newBenefitPrismGroupId: string;
    newEmploymentPayrollSettingId: string;
    newPtoPolicyId: string;
    newWorkLocationId: string;
    newPositionPublicId: string;
    newTeamId: number | null;
    newContractOid: string | null;
    resumeFromStep: string | null;
}

export interface TransferSignatureResponse {
    id: string;
    profilePublicId: string;
    role: string;
    agreementType: string;
    signedAt: Date | null;
}

export interface TransferRecordResponse {
    id: string;
    status: string;
    organizationId: number;
    requesterProfilePublicId: string;
    sourceLegalEntityPublicId: string;
    destinationLegalEntityPublicId: string;
    effectiveDate: string;
    agreementId: string | null;
    items: TransferItemResponse[];
    signatures: TransferSignatureResponse[];
    createdAt: Date;
    updatedAt: Date;
}

export interface GetTransfersBySourceEntityResponse {
    transfers: TransferRecordResponse[];
    cursor: string | null;
    hasMore: boolean;
}

export interface GetTransfersBySourceEntityOptions {
    cursor?: string;
    limit?: number;
}

@Service()
export default class EntityTransferService {
    async createTransfer(input: CreateTransferInput, transaction?: Transaction): Promise<PeoEmployeeTransfer> {
        log.info({
            message: 'Creating entity transfer',
            organizationId: input.organizationId,
            itemCount: input.items.length,
        });

        const transfer = await PeoEmployeeTransfer.create(
            {
                organizationId: input.organizationId,
                requesterProfilePublicId: input.requesterProfilePublicId,
                sourceLegalEntityPublicId: input.sourceLegalEntityPublicId,
                destinationLegalEntityPublicId: input.destinationLegalEntityPublicId,
                effectiveDate: input.effectiveDate,
                agreementId: input.agreementId || null,
                status: TransferStatus.PENDING_SIGNATURES,
            },
            { transaction },
        );

        const items: PeoEmployeeTransferItem[] = [];
        for (const item of input.items) {
            const createdItem = await PeoEmployeeTransferItem.create(
                {
                    organizationId: input.organizationId,
                    transferId: transfer.id,
                    baseContractOid: item.baseContractOid,
                    newBenefitPrismGroupId: item.newBenefitPrismGroupId,
                    newEmploymentPayrollSettingId: item.newEmploymentPayrollSettingId,
                    newPtoPolicyId: item.newPtoPolicyId,
                    newWorkLocationId: item.newWorkLocationId,
                    newPositionPublicId: item.newPositionPublicId,
                    newTeamId: item.newTeamId || null,
                    status: TransferItemStatus.PENDING,
                },
                { transaction },
            );
            items.push(createdItem);
        }

        log.info({
            message: 'Entity transfer created',
            transferId: transfer.id,
            itemIds: items.map(i => i.id),
        });

        // Return transfer with items
        return this.getTransferById(transfer.id, true, transaction) as Promise<PeoEmployeeTransfer>;
    }

    async getTransferById(id: string, includeItems = true, transaction?: Transaction): Promise<PeoEmployeeTransfer | null> {
        const include: any[] = [];

        if (includeItems) {
            include.push({ model: PeoEmployeeTransferItem, as: 'items' });
        }

        return PeoEmployeeTransfer.findByPk(id, {
            include,
            transaction,
            useMaster: !!transaction, // Use master if in transaction, replica otherwise
        });
    }

    async getTransferItemById(id: string, transaction?: Transaction): Promise<PeoEmployeeTransferItem | null> {
        return PeoEmployeeTransferItem.findByPk(id, {
            transaction,
            useMaster: !!transaction, // Use master if in transaction, replica otherwise
        });
    }

    async getTransferItemByIdWithTransfer(id: string, transaction?: Transaction): Promise<PeoEmployeeTransferItem | null> {
        return PeoEmployeeTransferItem.findByPk(id, {
            include: [
                {
                    model: PeoEmployeeTransfer,
                    as: 'transfer',
                    include: [{ model: PeoEmployeeTransferItem, as: 'items' }],
                },
            ],
            transaction,
            useMaster: !!transaction, // Use master if in transaction, replica otherwise
        });
    }

    async getReadyTransfers(effectiveDate: Date, limit = 100): Promise<PeoEmployeeTransfer[]> {
        return PeoEmployeeTransfer.findAll({
            where: {
                status: TransferStatus.SCHEDULED,
                effectiveDate: {
                    [Op.lte]: effectiveDate,
                },
            },
            include: [{ model: PeoEmployeeTransferItem, as: 'items' }],
            limit,
            order: [['effectiveDate', 'ASC']],
            useMaster: false, // Always use replica for batch queries
        });
    }

    async updateTransferStatus(id: string, status: TransferStatus, transaction?: Transaction): Promise<void> {
        log.info({ message: 'Updating transfer status', transferId: id, status });

        const [affectedCount] = await PeoEmployeeTransfer.update({ status }, { where: { id }, transaction });

        if (affectedCount === 0) {
            throw new Error(`Transfer not found: ${id}`);
        }
    }

    async updateTransferItem(id: string, updates: UpdateTransferItemInput, transaction?: Transaction): Promise<void> {
        log.info({ message: 'Updating transfer item', itemId: id, updates });

        const updateData: Partial<PeoEmployeeTransferItem> = {};

        if (updates.status !== undefined) {
            updateData.status = updates.status;
        }
        if (updates.newContractOid !== undefined) {
            updateData.newContractOid = updates.newContractOid;
        }
        if (updates.resumeFromStep !== undefined) {
            updateData.resumeFromStep = updates.resumeFromStep;
        }

        const [affectedCount] = await PeoEmployeeTransferItem.update(updateData, {
            where: { id },
            transaction,
        });

        if (affectedCount === 0) {
            throw new Error(`Transfer item not found: ${id}`);
        }
    }

    async updateTransferItemStatus(id: string, status: TransferItemStatus, transaction?: Transaction): Promise<void> {
        await this.updateTransferItem(id, { status }, transaction);
    }

    async getTransfersBySourceEntity(
        sourceEntityPublicId: string,
        options: GetTransfersBySourceEntityOptions = {},
    ): Promise<GetTransfersBySourceEntityResponse> {
        const { cursor, limit = 100 } = options;
        const queryLimit = Math.min(limit, 100);

        const whereClause: WhereOptions<PeoEmployeeTransfer> = {
            sourceLegalEntityPublicId: sourceEntityPublicId,
        };

        if (cursor) {
            whereClause.id = { [Op.gt]: cursor };
        }

        const transfers = await PeoEmployeeTransfer.findAll({
            where: whereClause,
            limit: queryLimit + 1,
            order: [['id', 'ASC']],
            include: [
                {
                    model: PeoEmployeeTransferItem,
                    as: 'items',
                },
                {
                    model: PeoEmployeeTransferSignature,
                    as: 'signatures',
                },
            ],
            useMaster: false,
        });

        const hasMore = transfers.length > queryLimit;
        const results = hasMore ? transfers.slice(0, queryLimit) : transfers;
        const nextCursor = hasMore && results.length > 0 ? results[results.length - 1].id : null;

        const baseContractOids = results.flatMap(t => t.items?.map(i => i.baseContractOid) || []);
        const contractsMap = await this.getContractsByOids(baseContractOids);

        const transformedTransfers = results.map(transfer => this.transformTransfer(transfer, contractsMap));

        return {
            transfers: transformedTransfers,
            cursor: nextCursor,
            hasMore,
        };
    }

    private async getContractsByOids(oids: string[]): Promise<Map<string, PeoContract>> {
        if (oids.length === 0) {
            return new Map();
        }

        const uniqueOids = [...new Set(oids)];
        const contracts = await PeoContract.findAll({
            where: {
                deelContractOid: { [Op.in]: uniqueOids },
            },
            attributes: ['deelContractOid', 'deelContractId', 'firstName', 'lastName'],
            useMaster: false,
        });

        const map = new Map<string, PeoContract>();
        for (const contract of contracts) {
            map.set(contract.deelContractOid, contract);
        }
        return map;
    }

    private transformTransfer(
        transfer: PeoEmployeeTransfer,
        contractsMap: Map<string, PeoContract>,
    ): TransferRecordResponse {
        return {
            id: transfer.id,
            status: transfer.status,
            organizationId: transfer.organizationId,
            requesterProfilePublicId: transfer.requesterProfilePublicId,
            sourceLegalEntityPublicId: transfer.sourceLegalEntityPublicId,
            destinationLegalEntityPublicId: transfer.destinationLegalEntityPublicId,
            effectiveDate: transfer.effectiveDate instanceof Date
                ? transfer.effectiveDate.toISOString().split('T')[0]
                : String(transfer.effectiveDate),
            agreementId: transfer.agreementId || null,
            items: transfer.items?.map(item => this.transformItem(item, contractsMap)) || [],
            signatures: transfer.signatures?.map(sig => this.transformSignature(sig)) || [],
            createdAt: transfer.createdAt,
            updatedAt: transfer.updatedAt,
        };
    }

    private transformItem(
        item: PeoEmployeeTransferItem,
        contractsMap: Map<string, PeoContract>,
    ): TransferItemResponse {
        const peoContract = contractsMap.get(item.baseContractOid);
        const employeeName = peoContract
            ? `${peoContract.firstName || ''} ${peoContract.lastName || ''}`.trim()
            : '';

        return {
            id: item.id,
            status: item.status,
            baseContractOid: item.baseContractOid,
            deelContractId: peoContract?.deelContractId || null,
            employeeName,
            newBenefitPrismGroupId: item.newBenefitPrismGroupId,
            newEmploymentPayrollSettingId: item.newEmploymentPayrollSettingId,
            newPtoPolicyId: item.newPtoPolicyId,
            newWorkLocationId: item.newWorkLocationId,
            newPositionPublicId: item.newPositionPublicId,
            newTeamId: item.newTeamId || null,
            newContractOid: item.newContractOid || null,
            resumeFromStep: item.resumeFromStep || null,
        };
    }

    private transformSignature(sig: PeoEmployeeTransferSignature): TransferSignatureResponse {
        return {
            id: sig.id,
            profilePublicId: sig.profilePublicId,
            role: sig.role,
            agreementType: sig.agreementType,
            signedAt: sig.signedAt || null,
        };
    }
}
